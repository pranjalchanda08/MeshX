33 """
34 Initialize the CodeGen class with product and model profiles in YAML format.
35
36 Args:
37 target_name (str): Name of the target product.
38 prod_profile (str): Path to the product profile YAML file.
39 model_profile (str): Path to the model profile YAML file.
40 """
41 self.max_el = 1
42 self.target = target_name
43 self.meshx_root = meshx_root
44 # Load product and model profiles from YAML files
45 with open(prod_profile) as pj:

51 self.pid = None

55 self.deps = {} # Dependency dictionary
56 self.__find_prod_from_target()
57 self.file_insert = file_insert
58 self.genfile_h = self.meshx_root + "/default/inc/meshx_config.h"
59 self.define_fmt = "\n#define {}{}{}"
60 if os.name == 'nt':
61 os.system(f'echo.>{self.genfile_h}') # Create the header file

63 os.system(f'touch {self.genfile_h}') # Create the header file

72 """
73 Get the define format for the autogenerated code.

78 def __find_prod_from_target(self):
79 """
80 Find the product from the target name.
81 """
82 self.product_list = self.prod_profile['prod']['products']
83 for product in self.product_list:
84 if product['name'] == self.target:
85 self.product = product
86 self.pid = self.product['pid']
87 if self.product is None:
88 raise CodeGenException("Product Not Found")

90 def __get_src_from_path(self, path):
91 """
92 Get all source files from the given path.
93 """
94 if path is None:
95 return
96 path_walk = self.meshx_root + "/" + path
97 for (root,dirs,files) in os.walk(path_walk):
98 for file in files:
99 if file.endswith('.c'): # Check if the file ends with .c
100 filepath = f"{root}/{file}"
101 if filepath not in self.c_list:
102 self.c_list.append(filepath) # Add full path to list
103 if file.endswith('.h') and f"{root}" not in self.inc_list:
104 self.inc_list.append(root)

106 def __resolve_dep_create_yml_dict(self, dep_s, value:int = 0):
107 """
108 Resolve dependencies and create a dictionary from the YAML file.
109 """
110 for _idx, dep in enumerate(self.prod_profile['elements']):
111 if dep["name"] == dep_s:
112 self.deps[dep_s] = {'path' : dep['path']}
113 if isinstance(dep['macro']['value'], bool):
114 self.prod_profile['elements'][_idx]['macro']['value'] = True
115 else:
116 self.prod_profile['elements'][_idx]['macro']['value'] = value
117
118 for sub_dep in dep['deps']:
119 self.__resolve_dep_create_yml_dict(sub_dep)

130 def get_common_defs(self):
131 """
132 Get common definitions from the product profile.
133 """
134 self.max_el = 1
135 if self.product is not None:
136 for element in self.product['elements']:
137 values = element.values()
138 for value in values:
139 self.max_el += value
140 self.file_insert += self.__get_fmt_define("CONFIG_CID_ID", hex(self.cid))
141 self.file_insert += self.__get_fmt_define("CONFIG_PID_ID", hex(self.pid))
142 self.file_insert += self.__get_fmt_define("CONFIG_PRODUCT_NAME", f'"{self.target[:16:]}"')
143 self.file_insert += self.__get_fmt_define("CONFIG_MAX_ELEMENT_COUNT", self.max_el)
144 return self.file_insert

146 def resolve_component_deps(self, component, value=0):
147 """
148 Resolve component dependencies and their dependencies.
149 """
150 if self.product is not None:
151 comp_ptr = self.prod_profile['components'][component]

157 def resolve_base_models_deps(self, base_model):

162 self.__get_src_from_path(res_base_model['path'])

164 macro['value'] = True

166 def resolve_default_models(self):

183 com['macros'][0]['value'] = True

186 def resolve_model_deps(self, model, val):
187 """
188 Resolve model dependencies and their dependencies.
189 """
190 res_model = self.prod_profile['models'][model]
191 self.__get_src_from_path(res_model['path'])
192 for macro in res_model['macros']:
193 if isinstance(macro['value'], bool):
194 macro['value'] = True
195 else:
196 macro['value'] += val
197 if res_model['base_model'] is not None:
198 self.resolve_base_models_deps(res_model['base_model'])

200 def resolve_el_deps(self, element, value=0):
201 """
202 Resolve element dependencies and their dependencies.
203 """
204 if self.product is not None:
205 element_ptr = self.prod_profile['elements'][element]
206 element_ptr['macros'][0]['value'] = value
207 self.__get_src_from_path(element_ptr['path'])
208 for model_reg in element_ptr['models']:
209 for model_key, model_val in model_reg.items():
210 self.resolve_model_deps(model_key, (model_val * value))

212 def resolve_prod_el(self):
213 """
214 Resolve product elements and their dependencies.
215 """
216 if self.product is not None:
217 for element in self.product['elements']:
218 self.resolve_el_deps(list(element.keys())[0], list(element.values())[0])
219 try:
220 for component in self.product['components']:
