54static esp_err_t ble_mesh_element_init(dev_struct_t *p_dev, meshx_config_t const *config)
55{
56 if (!p_dev)
57 return ESP_ERR_INVALID_STATE;
58
59 esp_err_t err = ESP_OK;
60 /* Initialize root model */
61 p_dev->elements[0].sig_models = get_root_sig_models();
62 p_dev->elements[0].vnd_models = ESP_BLE_MESH_MODEL_NONE;
63 memset((void *)&p_dev->elements[0].sig_model_count, get_root_sig_models_count(), sizeof(p_dev->elements[0].sig_model_count));
64 memset((void *)&p_dev->elements[0].vnd_model_count, ROOT_MODEL_VEN_CNT, sizeof(p_dev->elements[0].vnd_model_count));
65
66 ESP_LOGI(TAG, "Root: SIG : %d, VEN: %d", p_dev->elements[0].sig_model_count, p_dev->elements[0].vnd_model_count);
67 /* Root to be used with fixed format */
68 p_dev->element_idx++;
69
70 err = create_ble_mesh_element_composition(p_dev, config);
71 if(err)
72 {
73 ESP_LOGE(TAG, "Failed to create BLE Mesh Element Composition: (%d)", err);
74 return err;
75 }
76 p_dev->composition.cid = config->cid;
77 p_dev->composition.pid = config->pid;
78 p_dev->composition.element_count = p_dev->element_idx;
79 p_dev->composition.elements = p_dev->elements;
80
81 return ESP_OK;
82}

128static esp_err_t ble_mesh_init(meshx_config_t const *config)
129{
130 if(config == NULL || config->product_name == NULL || strlen(config->product_name) > ESP_BLE_MESH_DEVICE_NAME_MAX_LEN)
131 return ESP_ERR_INVALID_ARG;
132
133 esp_err_t err;
134
135 meshx_dev_restore(&g_dev, config);
136
137 err = ble_mesh_element_init(&g_dev, config);
138 ESP_ERR_PRINT_RET("Failed to initialize BLE Elements", err);
139
140 err = esp_ble_mesh_init(&MESHX_PROV_INSTANCE, &g_dev.composition);
141 ESP_ERR_PRINT_RET("Failed to initialize mesh stack", err);
142
143 err = esp_ble_mesh_set_unprovisioned_device_name(config->product_name);
144 ESP_ERR_PRINT_RET("Name Set Error", err);
145
146 err = esp_ble_mesh_node_prov_enable((esp_ble_mesh_prov_bearer_t)(ESP_BLE_MESH_PROV_ADV | ESP_BLE_MESH_PROV_GATT));
147 ESP_ERR_PRINT_RET("Failed to enable mesh node", err);
148
149 ESP_LOGI(TAG, "BLE Mesh Node initialized");
150
151 return ESP_OK;
152}

206esp_err_t meshx_init(meshx_config_t const *config)
207{
208 if(!config)
209 return ESP_ERR_INVALID_ARG;
210
211 esp_err_t err = ESP_OK;
212
213 err = nvs_flash_init();
214 if (err == ESP_ERR_NVS_NO_FREE_PAGES)
215 {
216 ESP_ERROR_CHECK(nvs_flash_erase());
217 err = nvs_flash_init();
218 }
219
220 ESP_ERROR_CHECK(err);
221
222 memcpy(&g_config, config, sizeof(meshx_config_t));
223
224 err = os_timer_init();
225 ESP_ERR_PRINT_RET("OS Timer Init failed", err);
226
227
228 err = meshx_nvs_init();
229 ESP_ERR_PRINT_RET("MeshX NVS Init failed", err);
230
231 err = meshx_tasks_init(&g_dev);
232 ESP_ERR_PRINT_RET("Tasks initialization failed", err);
233
234 err = meshx_init_boot_timer();
235 ESP_ERR_PRINT_RET("Boot Timer Init failed", err);
236
237 err = meshx_app_reg_element_callback(g_config.app_element_cb);
238 ESP_ERR_PRINT_RET("Failed to register app element callback", err);
239
240 err = meshx_app_reg_system_events_callback(g_config.app_ctrl_cb);
241 ESP_ERR_PRINT_RET("Failed to register app control callback", err);
242
243 err = bluetooth_init();
244 ESP_ERR_PRINT_RET("esp32_bluetooth_init failed", err);
245
246 esp_log_level_set("BLE_MESH", ESP_LOG_ERROR);
247
248 /* Initialize the Bluetooth Mesh Subsystem */
249 err = ble_mesh_init(&g_config);
250 ESP_ERR_PRINT_RET("Bluetooth mesh init failed", err);
251
252 printf(LOG_ANSI_COLOR_REGULAR(LOG_ANSI_COLOR_CYAN) "%s" LOG_ANSI_COLOR_RESET, meshX_banner);
253
254#if CONFIG_ENABLE_UNIT_TEST
255 err = register_ut_command();
256 ESP_ERR_PRINT_RET("Failed to register unit test command", err);
257
258 err = init_unit_test_console();
259 ESP_ERR_PRINT_RET("Failed to initialize production console", err);
260#endif /* CONFIG_ENABLE_UNIT_TEST */
261
262 return err;
263}
