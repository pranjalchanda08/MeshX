61static esp_err_t meshx_element_struct_init(uint16_t n_max)
62{
63 relay_element_init_ctrl = (relay_client_elements_t *)calloc(1, sizeof(relay_client_elements_t));
64 if (!relay_element_init_ctrl)
65 {
66 ESP_LOGE(TAG, "Failed to allocate memory for relay element structure");
67 return ESP_ERR_NO_MEM;
68 }
69 relay_element_init_ctrl->element_cnt = n_max;
70 relay_element_init_ctrl->element_id_end = 0;
71 relay_element_init_ctrl->element_id_start = 0;
72
73 relay_element_init_ctrl->rel_cli_ctx = (rel_cli_ctx_t *)calloc(n_max, sizeof(rel_cli_ctx_t));
74 if (!relay_element_init_ctrl->rel_cli_ctx)
75 {
76 ESP_LOGE(TAG, "Failed to allocate memory for relay client context");
77 return ESP_ERR_NO_MEM;
78 }
79 relay_element_init_ctrl->relay_cli_pub_list = (esp_ble_mesh_model_pub_t *)calloc(n_max, sizeof(esp_ble_mesh_model_pub_t));
80 if (!relay_element_init_ctrl->relay_cli_pub_list)
81 {
82 ESP_LOGE(TAG, "Failed to allocate memory for relay client pub list");
83 return ESP_ERR_NO_MEM;
84 }
85 relay_element_init_ctrl->relay_cli_onoff_gen_list = (esp_ble_mesh_client_t *)calloc(n_max, sizeof(esp_ble_mesh_client_t));
86 if (!relay_element_init_ctrl->relay_cli_onoff_gen_list)
87 {
88 ESP_LOGE(TAG, "Failed to allocate memory for relay client onoff gen list");
89 return ESP_ERR_NO_MEM;
90 }
91 relay_element_init_ctrl->relay_cli_sig_model_list = (esp_ble_mesh_model_t **)calloc(n_max, sizeof(esp_ble_mesh_model_t *));
92 if (!relay_element_init_ctrl->relay_cli_sig_model_list)
93 {
94 ESP_LOGE(TAG, "Failed to allocate memory for relay client sig model list");
95 return ESP_ERR_NO_MEM;
96 }
97 else
98 {
99 for (size_t i = 0; i < n_max; i++)
100 {
101 relay_element_init_ctrl->relay_cli_sig_model_list[i] = (esp_ble_mesh_model_t *)calloc(RELAY_CLI_MODEL_SIG_CNT, sizeof(esp_ble_mesh_model_t));
102 if (!relay_element_init_ctrl->relay_cli_sig_model_list[i])
103 {
104 ESP_LOGE(TAG, "Failed to allocate memory for relay client sig model list");
105 return ESP_ERR_NO_MEM;
106 }
107 }
108 }
109 return ESP_OK;
110}

164static esp_err_t meshx_dev_create_relay_model_space(dev_struct_t const *pdev, uint16_t n_max)
165{
166 if (!pdev)
167 return ESP_ERR_INVALID_STATE;
168
169 /* Assign Spaces for Model List, Publish List and onoff gen list */
170 esp_err_t err = meshx_element_struct_init(n_max);
171 if (err)
172 {
173 ESP_LOGE(TAG, "Failed to initialize relay element structures: (%d)", err);
174 meshx_element_struct_deinit(n_max);
175 return err;
176 }
177#if CONFIG_GEN_ONOFF_CLIENT_COUNT
178 for (size_t relay_model_id = 0; relay_model_id < n_max; relay_model_id++)
179 {
180 /* Perform memcpy to setup the constants */
181 memcpy(&relay_element_init_ctrl->relay_cli_sig_model_list[relay_model_id][0],
182 &relay_sig_template,
183 sizeof(esp_ble_mesh_model_t));
184 /* Set the dynamic spaces for the model */
185 void **temp = (void **)&relay_element_init_ctrl->relay_cli_sig_model_list[relay_model_id][0].pub;
186 *temp = relay_element_init_ctrl->relay_cli_pub_list + relay_model_id;
187 relay_element_init_ctrl->relay_cli_sig_model_list[relay_model_id][0].user_data =
188 relay_element_init_ctrl->relay_cli_onoff_gen_list + relay_model_id;
189 }
190#endif /* CONFIG_GEN_ONOFF_CLIENT_COUNT */
191 return ESP_OK;
192}

205static esp_err_t meshx_add_relay_cli_model_to_element_list(dev_struct_t *pdev, uint16_t *start_idx, uint16_t n_max)
206{
207 if (!pdev)
208 return ESP_ERR_INVALID_STATE;
209
210 if (!start_idx)
211 return ESP_ERR_INVALID_ARG;
212
213 if ((n_max + *start_idx) > CONFIG_MAX_ELEMENT_COUNT)
214 {
215 ESP_LOGE(TAG, "No of elements limit reached namx|start_idx|config_max: %d|%d|%d", n_max, *start_idx, CONFIG_MAX_ELEMENT_COUNT);
216 return ESP_ERR_NO_MEM;
217 }
218
219 esp_err_t err = ESP_OK;
220 esp_ble_mesh_elem_t *elements = pdev->elements;
221 relay_element_init_ctrl->element_id_start = *start_idx;
222
223 for (uint16_t i = *start_idx; i < (n_max + *start_idx); i++)
224 {
225 if (i == 0)
226 {
227 /* Insert the first SIG model in root model to save element virtual addr space */
228 memcpy(&elements[i].sig_models[1],
229 relay_element_init_ctrl->relay_cli_sig_model_list[i - *start_idx],
230 sizeof(esp_ble_mesh_model_t));
231 uint8_t *ref_ptr = (uint8_t *)&elements[i].sig_model_count;
232 (*ref_ptr)++;
233 }
234 else
235 {
236 ESP_LOGD(TAG, "Relay Client Element: %d", i);
237 elements[i].sig_models = relay_element_init_ctrl->relay_cli_sig_model_list[i - *start_idx];
238 elements[i].vnd_models = ESP_BLE_MESH_MODEL_NONE;
239 uint8_t *ref_ptr = (uint8_t *)&elements[i].sig_model_count;
240 *ref_ptr = RELAY_CLI_MODEL_SIG_CNT;
241 ref_ptr = (uint8_t *)&elements[i].vnd_model_count;
242 *ref_ptr = RELAY_CLI_MODEL_VEN_CNT;
243 }
244 err = meshx_nvs_elemnt_ctx_get(i, &(relay_element_init_ctrl->rel_cli_ctx[i - *start_idx]), sizeof(rel_cli_ctx_t));
245 if (err != ESP_OK)
246 {
247 ESP_LOGW(TAG, "Failed to get cwww cli element context: (0x%x)", err);
248 }
249 }
250 /* Increment the index for further registrations */
251 relay_element_init_ctrl->element_id_end = *start_idx += n_max;
252 return ESP_OK;
253}

265void meshx_relay_cli_generic_client_cb(const esp_ble_mesh_generic_client_cb_param_t *param, meshx_onoff_cli_evt_t evt)
266{
267 uint8_t element_id = param->params->model->element_idx;
268 if (!IS_EL_IN_RANGE(element_id))
269 {
270 return;
271 }
272
273 size_t rel_el_id = GET_RELATIVE_EL_IDX(element_id);
274 rel_cli_ctx_t *el_ctx = &relay_element_init_ctrl->rel_cli_ctx[rel_el_id];
275 relay_client_msg_t msg = {0};
276 meshx_el_relay_client_evt_t app_notify;
277 esp_err_t err;
278
279 switch (evt)
280 {
281 case MESHX_ONOFF_CLI_EVT_SET:
282 case MESHX_ONOFF_CLI_PUBLISH:
283 relay_element_init_ctrl->element_model_init |= BIT(RELAY_SIG_ONOFF_MODEL_ID);
284 if (el_ctx->state.prev_on_off != param->status_cb.onoff_status.present_onoff)
285 {
286 el_ctx->state.prev_on_off = param->status_cb.onoff_status.present_onoff;
287 app_notify.err_code = 0;
288 app_notify.on_off = el_ctx->state.prev_on_off;
289
290 err = meshx_send_msg_to_app(element_id,
291 MESHX_ELEMENT_TYPE_RELAY_CLIENT,
292 MESHX_ELEMENT_FUNC_ID_RELAY_SERVER_ONN_OFF,
293 sizeof(meshx_el_relay_client_evt_t),
294 &app_notify);
295 if (err != ESP_OK)
296 {
297 ESP_LOGE(TAG, "Failed to send relay state change message: (%d)", err);
298 }
299
300 el_ctx->state.on_off = !param->status_cb.onoff_status.present_onoff;
301 ESP_LOGD(TAG, "SET|PUBLISH: %d", param->status_cb.onoff_status.present_onoff);
302 ESP_LOGD(TAG, "Next state: %d", el_ctx->state.on_off);
303 }
304 break;
305 case MESHX_ONOFF_CLI_TIMEOUT:
306 ESP_LOGD(TAG, "Timeout");
307 msg.element_id = param->params->model->element_idx;
308 msg.set_get = RELAY_CLI_MSG_SET;
309 msg.ack = RELAY_CLI_MSG_ACK;
310 /*
311 * @warning: Possible loop case:
312 * 1. Relay Client sends a message to the server
313 * 2. Timeout occurs
314 * 3. #1 and #2 are repeated with no break in stetes.
315 */
316 control_task_msg_publish(CONTROL_TASK_MSG_CODE_TO_BLE, CONTROL_TASK_MSG_EVT_TO_BLE_SET_ON_OFF, &msg, sizeof(msg));
317
318 app_notify.err_code = 1;
319 err = meshx_send_msg_to_app(element_id,
320 MESHX_ELEMENT_TYPE_LIGHT_CWWW_CLIENT,
321 MESHX_ELEMENT_FUNC_ID_RELAY_SERVER_ONN_OFF,
322 sizeof(meshx_el_relay_client_evt_t),
323 &app_notify);
324 if (err != ESP_OK)
325 {
326 ESP_LOGE(TAG, "Failed to send relay state change message: (%d)", err);
327 }
328 break;
329 default:
330 ESP_LOGW(TAG, "Unhandled event: %d", evt);
331 break;
332 }
333}

347static void relay_client_config_srv_cb(const esp_ble_mesh_cfg_server_cb_param_t *param, config_evt_t evt)
348{
349 rel_cli_ctx_t *el_ctx = NULL;
350 size_t rel_el_id = 0;
351 uint16_t element_id = 0;
352 bool nvs_save = false;
353
354 ESP_LOGD(TAG, "EVT: %p", (void *)evt);
355 switch (evt)
356 {
357 case CONFIG_EVT_MODEL_APP_KEY_BIND:
358 element_id = param->value.state_change.mod_app_bind.element_addr - esp_ble_mesh_get_primary_element_address();
359 if (!IS_EL_IN_RANGE(element_id))
360 return;
361 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
362 el_ctx = &relay_element_init_ctrl->rel_cli_ctx[rel_el_id];
363 el_ctx->app_id = param->value.state_change.mod_app_bind.app_idx;
364 nvs_save = true;
365 break;
366 case CONFIG_EVT_MODEL_PUB_ADD:
367 case CONFIG_EVT_MODEL_PUB_DEL:
368 element_id = param->value.state_change.mod_pub_set.element_addr - esp_ble_mesh_get_primary_element_address();
369 if (!IS_EL_IN_RANGE(element_id))
370 return;
371 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
372 el_ctx = &relay_element_init_ctrl->rel_cli_ctx[rel_el_id];
373 el_ctx->pub_addr = evt == CONFIG_EVT_MODEL_PUB_ADD ? param->value.state_change.mod_pub_set.pub_addr
374 : ESP_BLE_MESH_ADDR_UNASSIGNED;
375 el_ctx->app_id = param->value.state_change.mod_pub_set.app_idx;
376 nvs_save = true;
377 ESP_LOGI(TAG, "PUB_ADD: %d, %d, 0x%x, 0x%x", element_id, rel_el_id, el_ctx->pub_addr, el_ctx->app_id);
378 break;
379 default:
380 break;
381 }
382 if (nvs_save)
383 {
384 esp_err_t err = meshx_nvs_elemnt_ctx_set(element_id, el_ctx, sizeof(rel_cli_ctx_t));
385 if (err != ESP_OK)
386 {
387 ESP_LOGE(TAG, "Failed to set relay element context: (%d)", err);
388 }
389 }
390}

520esp_err_t ble_mesh_send_relay_msg(dev_struct_t *pdev, uint16_t element_id, uint8_t set_get, uint8_t ack)
521{
522 if (!pdev || !IS_EL_IN_RANGE(element_id))
523 return ESP_ERR_INVALID_ARG;
524
525 esp_err_t err = ESP_OK;
526 esp_ble_mesh_elem_t *element = &pdev->elements[element_id];
527 esp_ble_mesh_model_t *model = &element->sig_models[0];
528
529 size_t rel_el_id = element_id - relay_element_init_ctrl->element_id_start;
530 rel_cli_ctx_t *el_ctx = &relay_element_init_ctrl->rel_cli_ctx[rel_el_id];
531
532 uint16_t opcode = ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_GET;
533 if (RELAY_CLI_MSG_SET == set_get)
534 {
535 opcode = ack ? ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_SET : ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_SET_UNACK;
536 }
537
538 ESP_LOGD(TAG, "OPCODE: %p", (void *)(uint32_t)opcode);
539
540 /* Send message to the relay client */
541 err = meshx_onoff_client_send_msg(
542 model,
543 opcode,
544 el_ctx->pub_addr,
545 pdev->meshx_store.net_key_id,
546 el_ctx->app_id,
547 el_ctx->state.on_off,
548 el_ctx->tid);
549 if (err)
550 {
551 ESP_LOGE(TAG, "Relay Client Send Message failed: (%d)", err);
552 }
553 else
554 {
555 el_ctx->tid++;
556 if(opcode == ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_SET_UNACK)
557 {
558 el_ctx->state.prev_on_off = el_ctx->state.on_off;
559 el_ctx->state.on_off = !el_ctx->state.on_off;
560 }
561 }
562 return err;
563}

574esp_err_t create_relay_client_elements(dev_struct_t *pdev, uint16_t element_cnt)
575{
576 esp_err_t err;
577
578 err = meshx_dev_create_relay_model_space(pdev, element_cnt);
579 if (err)
580 {
581 ESP_LOGE(TAG, "Relay Model space create failed: (%d)", err);
582 return err;
583 }
584
585 err = meshx_add_relay_cli_model_to_element_list(pdev, (uint16_t *)&pdev->element_idx, element_cnt);
586 if (err)
587 {
588 ESP_LOGE(TAG, "Relay Model add to element create failed: (%d)", err);
589 return err;
590 }
591
592 err = meshx_onoff_client_init();
593 if (err)
594 {
595 ESP_LOGE(TAG, "meshx_onoff_client_init failed: (%d)", err);
596 return err;
597 }
598
599#if RELAY_CLI_MESHX_ONOFF_ENABLE_CB
600 err = meshx_onoff_reg_cb(&meshx_relay_cli_generic_client_cb, RELAY_CLI_MESHX_ONOFF_CLI_CB_EVT_BMAP);
601 if (err)
602 {
603 ESP_LOGE(TAG, "Relay Model callback reg failed: (%d)", err);
604 return err;
605 }
606
607#if CONFIG_ENABLE_CONFIG_SERVER
608 err = meshx_config_server_cb_reg(&relay_client_config_srv_cb, CONFIG_SERVER_CB_MASK);
609 if (err)
610 {
611 ESP_LOGE(TAG, "Relay Model config server callback reg failed: (%d)", err);
612 return err;
613 }
614#endif /* CONFIG_ENABLE_CONFIG_SERVER */
615#if defined(__MESHX_CONTROL_TASK__)
616 err = control_task_msg_subscribe(
617 CONTROL_TASK_MSG_CODE_TO_BLE,
618 CONTROL_TASK_MSG_CODE_EVT_MASK,
619 (control_task_msg_handle_t)&relay_cli_control_task_msg_handle);
620 if (err)
621 {
622 ESP_LOGE(TAG, "control task callback reg failed: (%d)", err);
623 return err;
624 }
625 err = control_task_msg_subscribe(
626 CONTROL_TASK_MSG_CODE_SYSTEM,
627 CONTROL_TASK_MSG_EVT_SYSTEM_FRESH_BOOT,
628 (control_task_msg_handle_t)&relay_cli_freshboot_control_task_msg_handle);
629 if (err)
630 {
631 ESP_LOGE(TAG, "control task callback reg failed: (%d)", err);
632 return err;
633 }
634#endif /* __MESHX_CONTROL_TASK__ */
635#if CONFIG_ENABLE_UNIT_TEST
636 err = register_unit_test(MODULE_ID_ELEMENT_SWITCH_RELAY_CLIENT, &relay_cli_unit_test_cb_handler);
637 if (err)
638 {
639 ESP_LOGE(TAG, "unit_test reg failed: (%d)", err);
640 return err;
641 }
642#endif /* CONFIG_ENABLE_UNIT_TEST */
643#endif /* RELAY_CLI_MESHX_ONOFF_ENABLE_CB */
644
645 return ESP_OK;
646}
