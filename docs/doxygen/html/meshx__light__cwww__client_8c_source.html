67static void cwww_client_generic_client_cb(const esp_ble_mesh_generic_client_cb_param_t *param, meshx_onoff_cli_evt_t evt)
68{
69 uint8_t element_id = param->params->model->element_idx;
70 if (!IS_EL_IN_RANGE(element_id))
71 {
72 return;
73 }
74
75 size_t rel_el_id = GET_RELATIVE_EL_IDX(element_id);
76 cwww_cli_ctx_t *el_ctx = &cwww_client_element_init_ctrl->cwww_cli_ctx[rel_el_id];
77 cwww_client_msg_t msg = {0};
78 meshx_el_light_cwww_client_evt_t app_notify;
79 esp_err_t err;
80
81 switch (evt)
82 {
83 case MESHX_ONOFF_CLI_EVT_SET:
84 case MESHX_ONOFF_CLI_PUBLISH:
85 cwww_client_element_init_ctrl->element_model_init |= BIT(CWWW_CLI_SIG_ONOFF_MODEL_ID);
86 if (el_ctx->prev_state.on_off != param->status_cb.onoff_status.present_onoff)
87 {
88 el_ctx->prev_state.on_off = param->status_cb.onoff_status.present_onoff;
89 app_notify.err_code = 0;
90 app_notify.state_change.on_off.state = el_ctx->prev_state.on_off;
91
92 err = meshx_send_msg_to_app(element_id,
93 MESHX_ELEMENT_TYPE_LIGHT_CWWW_CLIENT,
94 MESHX_ELEMENT_FUNC_ID_LIGHT_CWWW_SERVER_ONN_OFF,
95 sizeof(meshx_el_light_cwww_client_evt_t),
96 &app_notify);
97 if (err != ESP_OK)
98 {
99 ESP_LOGE(TAG, "Failed to send CWWW state change message: (%d)", err);
100 }
101
102 el_ctx->state.on_off = !param->status_cb.onoff_status.present_onoff;
103 ESP_LOGD(TAG, "SET|PUBLISH: %d", param->status_cb.onoff_status.present_onoff);
104 ESP_LOGD(TAG, "Next state: %d", el_ctx->state.on_off);
105 }
106 break;
107 case MESHX_ONOFF_CLI_TIMEOUT:
108 ESP_LOGD(TAG, "Timeout");
109 msg.element_id = param->params->model->element_idx;
110 msg.set_get = RELAY_CLI_MSG_SET;
111 msg.ack = RELAY_CLI_MSG_ACK;
112 /*
113 * @warning: Possible loop case:
114 * 1. Relay Client sends a message to the server
115 * 2. Timeout occurs
116 * 3. #1 and #2 are repeated with no break in stetes.
117 */
118 control_task_msg_publish(CONTROL_TASK_MSG_CODE_TO_BLE, CONTROL_TASK_MSG_EVT_TO_BLE_SET_ON_OFF, &msg, sizeof(msg));
119
120 app_notify.err_code = 1;
121 err = meshx_send_msg_to_app(element_id,
122 MESHX_ELEMENT_TYPE_LIGHT_CWWW_CLIENT,
123 MESHX_ELEMENT_FUNC_ID_RELAY_SERVER_ONN_OFF,
124 sizeof(meshx_el_relay_client_evt_t),
125 &app_notify);
126 if (err != ESP_OK)
127 {
128 ESP_LOGE(TAG, "Failed to send relay state change message: (%d)", err);
129 }
130 break;
131 default:
132 ESP_LOGW(TAG, "Unhandled event: %d", evt);
133 break;
134 }

145static bool cwww_client_ctl_client_cb(const esp_ble_mesh_light_client_cb_param_t *param, light_ctl_cli_evt_t evt)
146{
147 uint8_t element_id = param->params->model->element_idx;
148 if (!IS_EL_IN_RANGE(element_id))
149 return false;
150
151 size_t rel_el_id = GET_RELATIVE_EL_IDX(element_id);
152 cwww_cli_ctx_t *el_ctx = &cwww_client_element_init_ctrl->cwww_cli_ctx[rel_el_id];
153 cwww_client_msg_t msg = {0};
154 meshx_el_light_cwww_client_evt_t app_notify;
155 esp_err_t err;
156 bool state_change = false;
157
158 if (param->params->opcode == ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_STATUS)
159 {
160 if (el_ctx->prev_ctl_state.lightness != param->status_cb.ctl_status.present_ctl_lightness
161 || el_ctx->prev_ctl_state.temperature != param->status_cb.ctl_status.present_ctl_temperature
162 )
163 {
164 el_ctx->prev_ctl_state.lightness = param->status_cb.ctl_status.present_ctl_lightness;
165 el_ctx->prev_ctl_state.temperature = param->status_cb.ctl_status.present_ctl_temperature;
166 state_change = true;
167 }
168 }
169 else if (param->params->opcode == ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS)
170 {
171 if(el_ctx->prev_ctl_state.delta_uv != param->status_cb.ctl_temperature_status.present_ctl_delta_uv
172 || el_ctx->prev_ctl_state.temperature != param->status_cb.ctl_temperature_status.present_ctl_temperature)
173 {
174 el_ctx->prev_ctl_state.delta_uv = param->status_cb.ctl_temperature_status.present_ctl_delta_uv;
175 el_ctx->prev_ctl_state.temperature = param->status_cb.ctl_temperature_status.present_ctl_temperature;
176 state_change = true;
177 }
178 }
179 else if ((param->params->opcode == ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS) && (param->status_cb.ctl_temperature_range_status.status_code == ESP_OK))
180 {
181 if (el_ctx->prev_ctl_state.temp_range_max != param->status_cb.ctl_temperature_range_status.range_max
182 || el_ctx->prev_ctl_state.temp_range_min != param->status_cb.ctl_temperature_range_status.range_min)
183 {
184 el_ctx->prev_ctl_state.temp_range_max = param->status_cb.ctl_temperature_range_status.range_max;
185 el_ctx->prev_ctl_state.temp_range_min = param->status_cb.ctl_temperature_range_status.range_min;
186 state_change = true;
187 }
188 }
189 else if (ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS == param->params->opcode)
190 {
191 if( el_ctx->prev_ctl_state.temp_def != param->status_cb.ctl_default_status.temperature
192 || el_ctx->prev_ctl_state.delta_uv_def != param->status_cb.ctl_default_status.delta_uv
193 || el_ctx->prev_ctl_state.lightness_def != param->status_cb.ctl_default_status.lightness)
194 {
195 el_ctx->prev_ctl_state.temp_def = param->status_cb.ctl_default_status.temperature;
196 el_ctx->prev_ctl_state.delta_uv_def = param->status_cb.ctl_default_status.delta_uv;
197 el_ctx->prev_ctl_state.lightness_def = param->status_cb.ctl_default_status.lightness;
198 state_change = true;
199 }
200 }
201 else
202 {
203 /* Return as No CTL related OPCode were received */
204 return false;
205 }
206
207 switch (evt)
208 {
209 case LIGHT_CTL_CLI_EVT_SET:
210 case LIGHT_CTL_CLI_PUBLISH:
211 cwww_client_element_init_ctrl->element_model_init |= BIT(CWWW_CLI_SIG_L_CTL_MODEL_ID);
212 if (state_change)
213 {
214 ESP_LOGD(TAG, "PUBLISH: light|temp : %d|%d",
215 el_ctx->prev_ctl_state.lightness,
216 el_ctx->prev_ctl_state.temperature);
217
218 app_notify.err_code = 0;
219 app_notify.state_change.ctl.delta_uv = el_ctx->prev_ctl_state.delta_uv;
220 app_notify.state_change.ctl.lightness = el_ctx->prev_ctl_state.lightness;
221 app_notify.state_change.ctl.temperature = el_ctx->prev_ctl_state.temperature;
222 app_notify.state_change.ctl.temp_range_max = el_ctx->prev_ctl_state.temp_range_max;
223 app_notify.state_change.ctl.temp_range_min = el_ctx->prev_ctl_state.temp_range_min;
224
225 err = meshx_send_msg_to_app(element_id,
226 MESHX_ELEMENT_TYPE_LIGHT_CWWW_CLIENT,
227 MESHX_ELEMENT_FUNC_ID_LIGHT_CWWW_CLIENT_CTL,
228 sizeof(meshx_el_light_cwww_client_evt_t),
229 &app_notify);
230 if (err != ESP_OK)
231 {
232 ESP_LOGE(TAG, "Failed to send CWWW state change message: (%d)", err);
233 }
234 }
235 break;
236 case LIGHT_CTL_CLI_TIMEOUT:
237 ESP_LOGD(TAG, "Timeout");
238 msg.ack = CWWW_CLI_MSG_ACK;
239 msg.set_get = CWWW_CLI_MSG_SET;
240 msg.element_id = param->params->model->element_idx;
241 /*
242 * @warning: Possible loop case:
243 * 1. CWWW Client sends a message to the server
244 * 2. Timeout occurs
245 * 3. #1 and #2 are repeated with no break in stetes.
246 */
247 control_task_msg_publish(CONTROL_TASK_MSG_CODE_TO_BLE, CONTROL_TASK_MSG_EVT_TO_BLE_SET_CTL, &msg, sizeof(msg));
248
249 app_notify.err_code = 1;
250 err = meshx_send_msg_to_app(element_id,
251 MESHX_ELEMENT_TYPE_LIGHT_CWWW_CLIENT,
252 MESHX_ELEMENT_FUNC_ID_LIGHT_CWWW_CLIENT_CTL,
253 sizeof(meshx_el_light_cwww_client_evt_t),
254 &app_notify);
255 if (err != ESP_OK)
256 {
257 ESP_LOGE(TAG, "Failed to send relay state change message: (%d)", err);
258 }
259 break;
260 default:
261 ESP_LOGW(TAG, "Unhandled event: %d", evt);
262 break;
263 }
264 return true;

333static esp_err_t cwww_cli_freshboot_control_task_msg_handle(const dev_struct_t *pdev, control_task_msg_evt_t evt, const void *params)
334{
335 if(!pdev)
336 return ESP_ERR_INVALID_ARG;
337
338 ESP_UNUSED(params);
339 ESP_UNUSED(evt);
340 cwww_client_msg_t msg = {0};
341
342 for(uint8_t i = CWWW_CLI_SIG_ONOFF_MODEL_ID; i < CWWW_CLI_SIG_ID_MAX; i++)
343 {
344 if(false == (cwww_client_element_init_ctrl->element_model_init & BIT(i)))
345 {
346 ESP_LOGD(TAG, "Sending GET for model: %d", i);
347 msg.ack = CWWW_CLI_MSG_ACK;
348 msg.set_get = CWWW_CLI_MSG_GET;
349 msg.element_id = (uint16_t)cwww_client_element_init_ctrl->element_id_start;
350 if(i == CWWW_CLI_SIG_L_CTL_MODEL_ID)
351 {
352 control_task_msg_publish(CONTROL_TASK_MSG_CODE_TO_BLE, CONTROL_TASK_MSG_EVT_TO_BLE_SET_CTL, &msg, sizeof(msg));
353 }
354 else
355 {
356 control_task_msg_publish(CONTROL_TASK_MSG_CODE_TO_BLE, CONTROL_TASK_MSG_EVT_TO_BLE_SET_ON_OFF, &msg, sizeof(msg));
357 }
358 }
359 }
360 return ESP_OK;
361}

395 el_ctx->ctl_state.temp_range_min = (msg->arg_bmap & CWWW_ARG_BMAP_TEMPERATURE_RANGE_SET_MIN) ? msg->temp_range_min : el_ctx->ctl_state.temp_range_min;

788static esp_err_t meshx_add_cwww_cli_model_to_element_list(dev_struct_t *pdev, uint16_t *start_idx, uint16_t n_max)
789{
790 if (!pdev || !start_idx)
791 {
792 return ESP_ERR_INVALID_ARG;
793 }
794
795 if ((n_max + *start_idx) > CONFIG_MAX_ELEMENT_COUNT)
796 {
797 ESP_LOGE(TAG, "No of elements limit reached namx|start_idx|config_max: %d|%d|%d", n_max, *start_idx, CONFIG_MAX_ELEMENT_COUNT);
798 return ESP_ERR_NO_MEM;
799 }
800 uint8_t *ref_ptr = NULL;
801 esp_err_t err = ESP_OK;
802 esp_ble_mesh_elem_t *elements = pdev->elements;
803
804 cwww_client_element_init_ctrl->element_id_start = *start_idx;
805
806 for (uint16_t i = *start_idx; i < (n_max + *start_idx) && (i - *start_idx) < CWWW_CLI_MODEL_SIG_CNT; i++)
807 {
808 if (i == 0)
809 {
810 /* Insert the first SIG model in root model to save element virtual addr space */
811 memcpy(&elements[i].sig_models[1],
812 cwww_client_element_init_ctrl->cwww_cli_sig_model_list[i - *start_idx],
813 sizeof(esp_ble_mesh_model_t));
814 ref_ptr = (uint8_t *)&elements[i].sig_model_count;
815 (*ref_ptr)++;
816 }
817 else
818 {
819 ESP_LOGD(TAG, "CWWW Client Element: %d", i);
820 elements[i].sig_models = cwww_client_element_init_ctrl->cwww_cli_sig_model_list[i - *start_idx];
821 elements[i].vnd_models = ESP_BLE_MESH_MODEL_NONE;
822 ref_ptr = (uint8_t *)&elements[i].sig_model_count;
823 *ref_ptr = CWWW_CLI_MODEL_SIG_CNT;
824 ref_ptr = (uint8_t *)&elements[i].vnd_model_count;
825 *ref_ptr = CWWW_CLI_MODEL_VEN_CNT;
826 }
827 err = meshx_nvs_elemnt_ctx_get(i, &(cwww_client_element_init_ctrl->cwww_cli_ctx[i - *start_idx]), sizeof(cwww_cli_ctx_t));
828 if (err != ESP_OK)
829 {
830 ESP_LOGW(TAG, "Failed to get cwww cli element context: (0x%x)", err);
831 }
832 }
833 /* Increment the index for further registrations */
834 cwww_client_element_init_ctrl->element_id_end = *start_idx += n_max;
835 return ESP_OK;
836}

944esp_err_t ble_mesh_send_cwww_msg(dev_struct_t *pdev, cwww_cli_sig_id_t model_id, uint16_t element_id, uint8_t set_get, uint8_t is_range, uint8_t ack)
945{
946 if (!pdev)
947 {
948 ESP_LOGE(TAG, "Invalid device structure");
949 return ESP_ERR_INVALID_ARG;
950 }
951
952 size_t rel_el_id = element_id - cwww_client_element_init_ctrl->element_id_start;
953 if (rel_el_id >= CONFIG_LIGHT_CWWW_CLIENT_COUNT)
954 {
955 ESP_LOGE(TAG, "Invalid element id: %d", element_id);
956 return ESP_ERR_INVALID_ARG;
957 }
958
959 uint16_t opcode = 0;
960 esp_err_t err = ESP_OK;
961 esp_ble_mesh_elem_t *element = &pdev->elements[element_id];
962 esp_ble_mesh_model_t *model = &element->sig_models[model_id];
963 cwww_cli_ctx_t *el_ctx = &cwww_client_element_init_ctrl->cwww_cli_ctx[rel_el_id];
964 light_ctl_send_args_t ctl_params = {0};
965
966 switch (model_id)
967 {
968 case CWWW_CLI_SIG_ONOFF_MODEL_ID:
969 if (set_get == CWWW_CLI_MSG_SET)
970 {
971 opcode = ack ? ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_SET : ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_SET_UNACK;
972 }
973 else
974 {
975 opcode = ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_GET;
976 }
977 ESP_LOGD(TAG, "OPCODE: %p", (void *)(uint32_t)opcode);
978 err = meshx_onoff_client_send_msg(model, opcode, el_ctx->pub_addr, pdev->meshx_store.net_key_id, el_ctx->app_id, el_ctx->state.on_off, el_ctx->tid);
979 if (!err)
980 {
981 el_ctx->tid++;
982 if (opcode == ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_SET_UNACK)
983 {
984 el_ctx->prev_state.on_off = el_ctx->state.on_off;
985 el_ctx->state.on_off = !el_ctx->state.on_off;
986 }
987 }
988 break;
989
990 case CWWW_CLI_SIG_L_CTL_MODEL_ID:
991 if (set_get == CWWW_CLI_MSG_SET)
992 {
993 if (is_range)
994 {
995 opcode = ack ? ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET : ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK;
996 ctl_params.temp_range_flag = true;
997 ctl_params.temp_range_max = el_ctx->ctl_state.temp_range_max;
998 ctl_params.temp_range_min = el_ctx->ctl_state.temp_range_min;
999 }
1000 else
1001 {
1002 opcode = ack ? ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_SET : ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_SET_UNACK;
1003 ctl_params.temp_range_flag = false;
1004 ctl_params.delta_uv = el_ctx->ctl_state.delta_uv;
1005 ctl_params.lightness = el_ctx->ctl_state.lightness;
1006 ctl_params.temperature = el_ctx->ctl_state.temperature;
1007 }
1008 }
1009 else
1010 {
1011 opcode = ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_GET;
1012 }
1013
1014 ctl_params.model = model;
1015 ctl_params.opcode = opcode;
1016 ctl_params.tid = el_ctx->tid;
1017 ctl_params.addr = el_ctx->pub_addr;
1018 ctl_params.app_idx = el_ctx->app_id;
1019 ctl_params.net_idx = pdev->meshx_store.net_key_id;
1020
1021 ESP_LOGD(TAG, "OPCODE: %p", (void *)(uint32_t)opcode);
1022 err = meshx_light_ctl_send_msg(&ctl_params);
1023 if (!err)
1024 {
1025 el_ctx->tid++;
1026 }
1027 break;
1028
1029 default:
1030 ESP_LOGE(TAG, "Invalid model id: %d", model_id);
1031 err = ESP_ERR_INVALID_ARG;
1032 break;
1033 }
1034
1035 if (err)
1036 {
1037 ESP_LOGE(TAG, "CWWW Client Send Message failed: (%d)", err);
1038 }
1039
1040 return err;
1041}
