128static esp_err_t meshx_prov_control_task_handler(dev_struct_t *pdev, control_task_msg_evt_t evt, void *params)
129{
130 const esp_ble_mesh_prov_cb_param_t *param = (esp_ble_mesh_prov_cb_param_t*) params;
131
132 switch (evt)
133 {
134 case CONTROL_TASK_MSG_EVT_PROVISION_STOP:
135 pdev->meshx_store.net_key_id = param->node_prov_complete.net_idx;
136 pdev->meshx_store.node_addr = param->node_prov_complete.addr;
137 meshx_nvs_set(MESHX_NVS_STORE, &pdev->meshx_store, sizeof(pdev->meshx_store), MESHX_NVS_AUTO_COMMIT);
138 break;
139 case CONTROL_TASK_MSG_EVT_IDENTIFY_START:
140 ESP_LOGI(TAG, "Identify Start");
141 break;
142 default:
143 break;
144 }
145 return ESP_OK;
146}

178esp_err_t create_ble_mesh_element_composition(dev_struct_t *p_dev, meshx_config_t const *config)
179{
180#if CONFIG_MAX_ELEMENT_COUNT > 0
181 esp_err_t err;
182#if CONFIG_SECTION_ENABLE_ELEMENT_TABLE
183 element_comp_table_t *element_comp_table;
184#endif /* CONFIG_SECTION_ENABLE_ELEMENT_TABLE */
185 if(!p_dev || !config || !config->element_comp_arr_len || !config->element_comp_arr)
186 return ESP_ERR_INVALID_ARG;
187
188 ble_mesh_get_dev_uuid(meshx_prov_cfg.uuid);
189
190 err = meshx_init_prov(&meshx_prov_cfg);
191 ESP_ERR_PRINT_RET("Failed to initialize Prov server", err);
192
193 err = control_task_msg_subscribe(
194 CONTROL_TASK_MSG_CODE_PROVISION,
195 CONTROL_TASK_PROV_EVT_MASK,
196 &meshx_prov_control_task_handler);
197 ESP_ERR_PRINT_RET("Failed to register control task callback", err);
198
199 err = meshx_init_config_server();
200 ESP_ERR_PRINT_RET("Failed to initialize config server", err);
201 for(uint16_t element_id = 0; element_id < config->element_comp_arr_len; element_id++)
202 {
203#if !CONFIG_SECTION_ENABLE_ELEMENT_TABLE
204 if(config->element_comp_arr[element_id].element_cnt != 0)
205 {
206 err = element_comp_fn[config->element_comp_arr[element_id].type](p_dev, config->element_comp_arr[element_id].element_cnt);
207 if(err)
208 {
209 ESP_LOGE(TAG, "Element composition failed: (%d)", err);
210 return err;
211 }
212 }
213#else
214 element_comp_table = &MESHX_ELEMENT_COMP_TABLE_START;
215 while(element_comp_table < &MESHX_ELEMENT_COMP_TABLE_STOP)
216 {
217 if(element_comp_table->idx == element_id && config->element_comp_arr[element_id].element_cnt != 0)
218 {
219 err = element_comp_table->element_comp_fn(p_dev, config->element_comp_arr[element_id].element_cnt);
220 if(err)
221 {
222 ESP_LOGE(TAG, "Element composition failed: (%d)", err);
223 return err;
224 }
225 }
226 element_comp_table++;
227 }
228#endif /* CONFIG_SECTION_ENABLE_ELEMENT_TABLE */
229 }
230#endif /* CONFIG_MAX_ELEMENT_COUNT */
231 return err;
232}
