44static void meshx_onoff_reg_cb_dispatch(const esp_ble_mesh_generic_client_cb_param_t *param, meshx_onoff_cli_evt_t evt)
45{
46 if (SLIST_EMPTY(&meshx_onoff_cli_cb_reg_table))
47 {
48 ESP_LOGW(TAG, "No onoff client callback registered for event: %p", (void *)evt);
49 return;
50 }
51
52 if (xSemaphoreTake(meshx_onoff_cli_mutex, portMAX_DELAY) == pdTRUE)
53 {
54 meshx_onoff_cli_cb_reg_t *ptr;
55 SLIST_FOREACH(ptr, &meshx_onoff_cli_cb_reg_table, entries)
56 {
57 if ((evt & ptr->evt_bmap) && ptr->cb != NULL)
58 {
59 ptr->cb(param, evt); // Call the registered callback
60 }
61 }
62 xSemaphoreGive(meshx_onoff_cli_mutex);
63 }
