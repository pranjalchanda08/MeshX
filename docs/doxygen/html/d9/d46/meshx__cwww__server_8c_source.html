78 element_id = param->value.state_change.mod_pub_set.element_addr - esp_ble_mesh_get_primary_element_address();
79 if (!IS_EL_IN_RANGE(element_id))

85 el_ctx->app_id = param->value.state_change.mod_pub_set.app_idx;

87 ESP_LOGI(TAG, "PUB_ADD: %d, %d, 0x%x, 0x%x", element_id, rel_el_id, el_ctx->pub_addr, el_ctx->app_id);

89 default:
90 break;
91 }
92 if(nvs_save)
93 {
94 esp_err_t err = meshx_nvs_elemnt_ctx_set(element_id, el_ctx, sizeof(cwww_server_ctx_t));

97 ESP_LOGE(TAG, "Failed to set cwww server element context: (%d)", err);
98 }
99 }

350static esp_err_t meshx_add_cwww_srv_model_to_element_list(dev_struct_t *pdev, uint16_t *start_idx, uint16_t n_max)
351{
352 if (!pdev)
353 return ESP_ERR_INVALID_STATE;
354
355 if (!start_idx || (n_max + *start_idx) > CONFIG_MAX_ELEMENT_COUNT)
356 {
357 ESP_LOGE(TAG, "No of elements limit reached");
358 return ESP_ERR_NO_MEM;
359 }
360 esp_err_t err = ESP_OK;
361 uint8_t *ref_ptr = NULL;
362 esp_ble_mesh_elem_t *elements = pdev->elements;
363 cwww_element_init_ctrl.element_id_start = *start_idx;
364
365 for (uint16_t i = *start_idx; i < (n_max + *start_idx); i++)
366 {
367 if (i == 0)
368 {
369 /* Insert the first SIG model in root model to save element virtual addr space */
370 memcpy(&elements[i].sig_models[1],
371 cwww_element_init_ctrl.cwww_server_sig_model_list[i - *start_idx],
372 sizeof(esp_ble_mesh_model_t));
373 ref_ptr = (uint8_t *)&elements[i].sig_model_count;
374 (*ref_ptr)++;
375 }
376 else
377 {
378 elements[i].sig_models = cwww_element_init_ctrl.cwww_server_sig_model_list[i - *start_idx];
379 elements[i].vnd_models = ESP_BLE_MESH_MODEL_NONE;
380 ref_ptr = (uint8_t *)&elements[i].sig_model_count;
381 *ref_ptr = CWWW_SRV_MODEL_SIG_CNT;
382 ref_ptr = (uint8_t *)&elements[i].vnd_model_count;
383 *ref_ptr = CWWW_SRV_MODEL_VEN_CNT;
384 }
385
386 err = meshx_nvs_elemnt_ctx_get(i, &(cwww_element_init_ctrl.cwww_server_ctx[i - *start_idx]), sizeof(cwww_server_ctx_t));
387 if (err != ESP_OK)
388 {
389 ESP_LOGW(TAG, "Failed to get cwww element context: (0x%x)", err);
390 }
391 else
392 {
393 err = meshx_restore_model_states(i - *start_idx);
394 if (err != ESP_OK)
395 {
396 ESP_LOGW(TAG, "Failed to restore cwww model states: (0x%x)", err);
397 }
398 }
399 }
400 /* Increment the index for further registrations */
401 cwww_element_init_ctrl.element_id_end = *start_idx += n_max;
402 return ESP_OK;
403}

425static esp_err_t meshx_el_control_task_handler(dev_struct_t const *pdev, control_task_msg_evt_t evt, void const *params)
426{
427 ESP_UNUSED(pdev);
428 esp_err_t err = ESP_OK;
429 uint16_t element_id = 0;
430 size_t rel_el_id;
431 cwww_server_ctx_t *el_ctx = NULL;
432 meshx_el_light_cwww_server_evt_t app_msg;
433 cwww_sig_id_t sig_func = CWWW_SIG_ONOFF_MODEL_ID;
434
435 switch (evt) {
436 case CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_ON_OFF:
437 {
438 const esp_ble_mesh_gen_onoff_srv_t *p_onoff_srv = (const esp_ble_mesh_gen_onoff_srv_t *)params;
439 element_id = p_onoff_srv->model->element_idx;
440 if (!IS_EL_IN_RANGE(element_id))
441 goto el_ctrl_task_hndlr_exit;
442
443 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
444 el_ctx = &cwww_element_init_ctrl.cwww_server_ctx[rel_el_id];
445 if(el_ctx->prev_state.on_off == p_onoff_srv->state.onoff)
446 goto el_ctrl_task_hndlr_exit;
447
448 el_ctx->prev_state.on_off = p_onoff_srv->state.onoff;
449 sig_func = CWWW_SIG_ONOFF_MODEL_ID;
450 app_msg.state_change.on_off.state = el_ctx->prev_state.on_off;
451 break;
452 }
453 case CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_CTL:
454 {
455 const esp_ble_mesh_light_ctl_srv_t *p_ctl_srv = (const esp_ble_mesh_light_ctl_srv_t *)params;
456 element_id = p_ctl_srv->model->element_idx;
457 if (!IS_EL_IN_RANGE(element_id)) goto el_ctrl_task_hndlr_exit;
458 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
459 el_ctx = &cwww_element_init_ctrl.cwww_server_ctx[rel_el_id];
460
461 if(el_ctx->prev_ctl_state.delta_uv == p_ctl_srv->state->delta_uv &&
462 el_ctx->prev_ctl_state.lightness == p_ctl_srv->state->lightness &&
463 el_ctx->prev_ctl_state.temperature == p_ctl_srv->state->temperature &&
464 el_ctx->prev_ctl_state.temp_range_min == p_ctl_srv->state->temperature_range_min &&
465 el_ctx->prev_ctl_state.temp_range_max == p_ctl_srv->state->temperature_range_max)
466 goto el_ctrl_task_hndlr_exit;
467
468 el_ctx->prev_ctl_state.delta_uv = p_ctl_srv->state->delta_uv;
469 el_ctx->prev_ctl_state.lightness = p_ctl_srv->state->lightness;
470 el_ctx->prev_ctl_state.temperature = p_ctl_srv->state->temperature;
471 el_ctx->prev_ctl_state.temp_range_min = p_ctl_srv->state->temperature_range_min;
472 el_ctx->prev_ctl_state.temp_range_max = p_ctl_srv->state->temperature_range_max;
473
474 sig_func = CWWW_SIG_L_CTL_MODEL_ID;
475 app_msg.state_change.ctl.delta_uv = el_ctx->prev_ctl_state.delta_uv;
476 app_msg.state_change.ctl.lightness = el_ctx->prev_ctl_state.lightness;
477 app_msg.state_change.ctl.temperature = el_ctx->prev_ctl_state.temperature;
478 app_msg.state_change.ctl.temp_range_min = el_ctx->prev_ctl_state.temp_range_min;
479 app_msg.state_change.ctl.temp_range_max = el_ctx->prev_ctl_state.temp_range_max;
480 break;
481 }
482 default:
483 break;
484 }
485
486 err = meshx_nvs_elemnt_ctx_set(element_id, el_ctx, sizeof(cwww_server_ctx_t));
487 if (err != ESP_OK)
488 ESP_LOGE(TAG, "Failed to set relay element context: (%d)", err);
489
490 err = meshx_send_msg_to_app(element_id, MESHX_ELEMENT_TYPE_LIGHT_CWWW_SERVER, (uint16_t) sig_func, sizeof(meshx_el_light_cwww_server_evt_t), &app_msg);
491 if (err != ESP_OK)
492 ESP_LOGE(TAG, "Failed to send relay state change message: (%d)", err);
493
494el_ctrl_task_hndlr_exit:
495 return ESP_OK;
496}
