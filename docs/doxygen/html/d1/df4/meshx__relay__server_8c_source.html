65static void relay_server_config_srv_cb(const esp_ble_mesh_cfg_server_cb_param_t *param, config_evt_t evt)
66{
67 relay_srv_model_ctx_t *el_ctx = NULL;
68 size_t rel_el_id = 0;
69 uint16_t element_id = 0;
70
71 ESP_LOGD(TAG, "EVT: %p", (void *)evt);
72 switch (evt)
73 {
74 case CONFIG_EVT_MODEL_APP_KEY_BIND:
75 element_id = (param->value.state_change.mod_app_bind.element_addr - esp_ble_mesh_get_primary_element_address());

77 break;
78 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
79 el_ctx = &relay_element_init_ctrl.meshx_gen_ctx[rel_el_id];
80 el_ctx->app_id = param->value.state_change.appkey_add.app_idx;
81 break;
82 case CONFIG_EVT_MODEL_PUB_ADD:
83 case CONFIG_EVT_MODEL_PUB_DEL:
84 element_id = (param->value.state_change.mod_pub_set.element_addr - esp_ble_mesh_get_primary_element_address());
85 if (!IS_EL_IN_RANGE(element_id))
86 break;
87 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
88 el_ctx = &relay_element_init_ctrl.meshx_gen_ctx[rel_el_id];
89 el_ctx->pub_addr = evt == CONFIG_EVT_MODEL_PUB_ADD ? param->value.state_change.mod_pub_set.pub_addr
90 : ESP_BLE_MESH_ADDR_UNASSIGNED;
91 el_ctx->app_id = param->value.state_change.mod_pub_set.app_idx;
92 ESP_LOGI(TAG, "PUB_ADD: %d, %d, 0x%x, 0x%x", element_id, rel_el_id, el_ctx->pub_addr, el_ctx->app_id);
93 break;
94 default:
95 break;
96 }
97}

272static esp_err_t meshx_add_relay_srv_model_to_element_list(dev_struct_t *pdev, uint16_t *start_idx, uint16_t n_max)
273{
274 if (!pdev)
275 return ESP_ERR_INVALID_STATE;
276
277 if ((n_max + *start_idx) > CONFIG_MAX_ELEMENT_COUNT)
278 {
279 ESP_LOGE(TAG, "No of elements limit reached");
280 return ESP_ERR_NO_MEM;
281 }
282
283 esp_err_t err = ESP_OK;
284 esp_ble_mesh_elem_t *elements = pdev->elements;
285 relay_element_init_ctrl.element_id_start = *start_idx;
286
287 for (uint16_t i = *start_idx; i < (n_max + *start_idx); i++)
288 {
289 if (i == 0)
290 {
291 memcpy(&elements[i].sig_models[1],
292 relay_element_init_ctrl.relay_server_sig_model_list[i],
293 sizeof(esp_ble_mesh_model_t));
294 uint8_t *ref_ptr = (uint8_t *)&elements[i].sig_model_count;
295 (*ref_ptr)++;
296 }
297 else
298 {
299 elements[i].sig_models = relay_element_init_ctrl.relay_server_sig_model_list[i - *start_idx];
300 elements[i].vnd_models = ESP_BLE_MESH_MODEL_NONE; // No vendor models are assigned to this element
301 uint8_t *ref_ptr = (uint8_t *)&elements[i].sig_model_count;
302 // Set the number of SIG models for the relay server element
303 *ref_ptr = RELAY_SRV_MODEL_SIG_CNT;
304 ref_ptr = (uint8_t *)&elements[i].vnd_model_count;
305 *ref_ptr = RELAY_SRV_MODEL_VEN_CNT;
306 }
307 err = meshx_nvs_elemnt_ctx_get(i, &(relay_element_init_ctrl.meshx_gen_ctx[i - *start_idx]), sizeof(relay_element_init_ctrl.meshx_gen_ctx[i]));
308 if (err != ESP_OK)
309 {
310 ESP_LOGW(TAG, "Failed to get relay element context: (0x%x)", err);
311 }
312 else
313 {
314 err = meshx_restore_model_states(i - *start_idx);
315 if (err != ESP_OK)
316 {
317 ESP_LOGW(TAG, "Failed to restore relay model states: (0x%x)", err);
318 }
319 }
320 }
321 relay_element_init_ctrl.element_id_end = (*start_idx += n_max);
322 return ESP_OK;
323}

335static esp_err_t meshx_el_control_task_handler(dev_struct_t const *pdev, control_task_msg_evt_t evt, void *params)
336{
337 ESP_UNUSED(pdev);
338 ESP_UNUSED(evt);
339 size_t rel_el_id = 0;
340 esp_err_t err = ESP_OK;
341 relay_srv_model_ctx_t *el_ctx = NULL;
342 esp_ble_mesh_gen_onoff_srv_t const *p_onoff_srv = (esp_ble_mesh_gen_onoff_srv_t*) params;
343 meshx_el_relay_server_evt_t state;
344 uint16_t element_id = p_onoff_srv->model->element_idx;
345
346 if (!IS_EL_IN_RANGE(element_id))
347 return ESP_OK;
348
349 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
350 el_ctx = &relay_element_init_ctrl.meshx_gen_ctx[rel_el_id];
351
352 p_onoff_srv = (esp_ble_mesh_gen_onoff_srv_t const *) params;
353 el_ctx->state = p_onoff_srv->state.onoff;
354
355 err = meshx_nvs_elemnt_ctx_set(element_id, el_ctx, sizeof(relay_srv_model_ctx_t));
356 if (err != ESP_OK)
357 ESP_LOGE(TAG, "Failed to set relay element context: (%d)", err);
358
359 state.on_off = el_ctx->state;
360 err = meshx_send_msg_to_app(element_id, MESHX_ELEMENT_TYPE_RELAY_SERVER, RELAY_SIG_ONOFF_MODEL_ID, sizeof(meshx_el_relay_server_evt_t), &state);
361 if (err != ESP_OK)
362 ESP_LOGE(TAG, "Failed to send relay state change message: (%d)", err);
363
364 return ESP_OK;
365}

375static esp_err_t relay_prov_control_task_handler(dev_struct_t const *pdev, control_task_msg_evt_t evt, void const *params)
376{
377 ESP_UNUSED(pdev);
378 ESP_UNUSED(evt);
379 ESP_UNUSED(params);
380
381 size_t rel_el_id = 0;
382 esp_ble_mesh_msg_ctx_t ctx;
383 esp_err_t err = ESP_OK;
384
385 for(size_t el_id = relay_element_init_ctrl.element_id_start; el_id < relay_element_init_ctrl.element_id_end; el_id++)
386 {
387 rel_el_id = GET_RELATIVE_EL_IDX(el_id);
388 ctx.net_idx = pdev->meshx_store.net_key_id;
389 ctx.app_idx = relay_element_init_ctrl.meshx_gen_ctx[rel_el_id].app_id;
390 ctx.addr = relay_element_init_ctrl.meshx_gen_ctx[rel_el_id].pub_addr;
391 ctx.send_ttl = ESP_BLE_MESH_TTL_DEFAULT;
392 ctx.send_cred = 0;
393 ctx.send_tag = BIT(1);
394
395 if(ctx.addr == ESP_BLE_MESH_ADDR_UNASSIGNED || ctx.app_idx == ESP_BLE_MESH_KEY_UNUSED)
396 continue;
397 err = esp_ble_mesh_server_model_send_msg(&relay_element_init_ctrl.relay_server_sig_model_list[rel_el_id][RELAY_SIG_ONOFF_MODEL_ID],
398 &ctx,
399 ESP_BLE_MESH_MODEL_OP_GEN_ONOFF_STATUS,
400 sizeof(relay_element_init_ctrl.meshx_gen_ctx[rel_el_id].state),
401 &relay_element_init_ctrl.meshx_gen_ctx[rel_el_id].state);
402 if (err)
403 {
404 ESP_LOGE(TAG, "Failed to send ONOFF status message (Err: %x)", err);
405 return err;
406 }
407 }
408
409 return ESP_OK;
