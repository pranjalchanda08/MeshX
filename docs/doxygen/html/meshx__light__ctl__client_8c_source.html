84esp_err_t meshx_light_ctl_cli_reg_cb(light_cli_cb cb, uint32_t config_evt_bmap)

181 if (err)
182 {
183 ESP_LOGE(TAG, "Send Generic OnOff failed");
184 return err;
185 }
186 }
187 if(send_msg)
188 {
189 err = esp_ble_mesh_light_client_set_state(&common, &set);
190 if (err)
191 {
192 ESP_LOGE(TAG, "Light CTL Client Send Message failed: (%d)", err);
