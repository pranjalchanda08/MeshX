80static void meshx_config_server_cb_dispatch(const esp_ble_mesh_cfg_server_cb_param_t *param, config_evt_t evt)
81{
82 if (SLIST_EMPTY(&config_server_cb_reg_table))
83 {
84 ESP_LOGW(TAG, "No config server callback registered for event: %p", (void *)evt);
85 return;
86 }
87
88 config_server_cb_reg_t *ptr;
89
90 SLIST_FOREACH(ptr, &config_server_cb_reg_table, next)
91 {
92 if ((evt & ptr->evt_bmap) && ptr->cb != NULL)
93 {
94 ptr->cb(param, evt); // Call the registered callback
95 }
96 }
97}

108static void meshx_ble_mesh_config_server_cb(esp_ble_mesh_cfg_server_cb_event_t event,
109 const esp_ble_mesh_cfg_server_cb_param_t *param)
110{
111 if (event == ESP_BLE_MESH_CFG_SERVER_STATE_CHANGE_EVT)
112 {
113 for (size_t map_id = 0; map_id < ARRAY_SIZE(config_server_model_evt_map_table); map_id++)
114 {
115 if (config_server_model_evt_map_table[map_id].model_op_code == param->ctx.recv_op)
116 {
117 ESP_LOGI(TAG, "%s", config_server_model_evt_map_table[map_id].op_str);
118 meshx_config_server_cb_dispatch(param, config_server_model_evt_map_table[map_id].config_evt);
119 break;
120 }
121 }
122 }

154esp_err_t meshx_config_server_cb_reg(config_srv_cb cb, uint32_t config_evt_bmap)
155{
156 if (cb == NULL || config_evt_bmap == 0)
157 {
158 return ESP_ERR_INVALID_ARG; // Invalid arguments
159 }
160
161 if (xSemaphoreTake(config_server_mutex, portMAX_DELAY) == pdTRUE)
162 {
163 config_server_cb_reg_t *new_node = (config_server_cb_reg_t *)malloc(sizeof(config_server_cb_reg_t));
164 if (new_node == NULL)
165 {
166 xSemaphoreGive(config_server_mutex);
167 return ESP_ERR_NO_MEM; // Memory allocation failed
168 }
169
170 new_node->cb = cb;
171 new_node->evt_bmap = config_evt_bmap;
172 SLIST_INSERT_HEAD(&config_server_cb_reg_table, new_node, next);
173
174 xSemaphoreGive(config_server_mutex);
175 }
176
177 return ESP_OK;
178}
