119 bool send_reply_to_src = false;
120 uint8_t ctl_status_pack_len = 0;

122 esp_err_t err = ESP_OK;
123
124 ctl_status_t ctl_status_union;
125
126 switch (op_code)
127 {
129 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_GET:
130 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_SET:

132 status_op = ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_STATUS;
133 if(op_code != ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_GET){
134 srv->state->temperature = param->value.state_change.ctl_set.temperature;
135 srv->state->lightness = param->value.state_change.ctl_set.lightness;
136 srv->state->delta_uv = param->value.state_change.ctl_set.delta_uv;

147
148 ctl_status_union.ctl_status.temperature = srv->state->temperature;
149 ctl_status_union.ctl_status.lightness = srv->state->lightness;
150 ctl_status_pack_len = sizeof(ctl_status_union.ctl_status);
151
152 break;

154 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET:
155 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET_UNACK:

177 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_DEFAULT_GET:

181 ESP_LOGI(TAG, "lightness|temp|del_uv:%d|%d|%d",

186 srv->state->lightness_default = param->value.state_change.ctl_default_set.lightness;
187 srv->state->delta_uv_default = param->value.state_change.ctl_default_set.delta_uv;
188 }

190 send_reply_to_src = true;
191
192 ctl_status_union.ctl_default.delta_uv_def = srv->state->delta_uv_default;
193 ctl_status_union.ctl_default.lightness_def = srv->state->lightness_default;
194 ctl_status_union.ctl_default.temperature_def = srv->state->temperature_default;
195 ctl_status_pack_len = sizeof(ctl_status_union.ctl_default);
196
197 status_op = ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS;
198 break;
199 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET:
200 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET:
201 case ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK:
202 if(op_code != ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET)
203 {
204 ESP_LOGI(TAG, "temp min|max: %dK|%dK",
205 param->value.state_change.ctl_temp_range_set.range_min,
206 param->value.state_change.ctl_temp_range_set.range_max);
207 srv->state->temperature_range_min = param->value.state_change.ctl_temp_range_set.range_min;
208 srv->state->temperature_range_max = param->value.state_change.ctl_temp_range_set.range_max;
209 }
210 if(op_code != ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK)
211 {
212 send_reply_to_src = true;
213 }
214
215 ctl_status_union.ctl_temp_range.status_code = ESP_OK;
216 ctl_status_union.ctl_temp_range.range_min = srv->state->temperature_range_min;
217 ctl_status_union.ctl_temp_range.range_max = srv->state->temperature_range_max;
218 ctl_status_pack_len = sizeof(ctl_status_union.ctl_temp_range);

220 status_op = ESP_BLE_MESH_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS;
221 break;
222 default:
223 ESP_LOGW(TAG, "CTL Unhandled Event %p", (void*)param->ctx.recv_op);
224 break;
225 }
226 if (send_reply_to_src
227 /* This is meant to notify the respective publish client */
228 || param->ctx.addr != param->model->pub->publish_addr)
229 {
230 /* Here the message was received from unregistered source and mention the state to the respective client */
231 ESP_LOGD(TAG, "PUB: src|pub %x|%x", param->ctx.addr, param->model->pub->publish_addr);
232 /* ACK to Source */
233 err = esp_ble_mesh_server_model_send_msg(param->model,
234 &param->ctx,
235 status_op,
236 ctl_status_pack_len,
237 (uint8_t*)&ctl_status_union);
238 if(err)
239 ESP_LOGE(TAG, "Failed to send CTL status message (Err: %x)", err);
240 }
241
242 return err;
