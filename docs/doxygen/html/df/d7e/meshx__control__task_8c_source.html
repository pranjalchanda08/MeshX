98esp_err_t control_task_msg_subscribe(control_task_msg_code_t msg_code,
99 control_task_msg_evt_t evt_bmap,
100 control_task_msg_handle_t callback)
101{
102 if (callback == NULL || evt_bmap == 0 || msg_code >= CONTROL_TASK_MSG_CODE_MAX)
103 return ESP_ERR_INVALID_ARG; // Invalid arguments
104
105 control_task_evt_cb_reg_t *new_node = (control_task_evt_cb_reg_t *)malloc(sizeof(control_task_evt_cb_reg_t));
106 if (new_node == NULL)
107 return ESP_ERR_NO_MEM; // Memory allocation failed
108
109 new_node->cb = callback;
110 new_node->msg_evt_bmap = evt_bmap;
111 new_node->next = control_task_msg_code_list_heads[msg_code];
112 control_task_msg_code_list_heads[msg_code] = new_node;
113
114 return ESP_OK;
115}

127esp_err_t control_task_msg_unsubscribe(control_task_msg_code_t msg_code,
128 control_task_msg_evt_t evt_bmap,
129 control_task_msg_handle_t callback)
130{
131 if (callback == NULL || evt_bmap == 0 || msg_code >= CONTROL_TASK_MSG_CODE_MAX)
132 return ESP_ERR_INVALID_ARG; // Invalid arguments
133
134 control_task_evt_cb_reg_t *prev = NULL;
135 control_task_evt_cb_reg_t *curr = control_task_msg_code_list_heads[msg_code];
136
137 while (curr) {
138 if (curr->cb == callback && curr->msg_evt_bmap == evt_bmap) {
139 if (prev == NULL) {
140 control_task_msg_code_list_heads[msg_code] = curr->next;
141 } else {
142 prev->next = curr->next;
143 }
144 free(curr);
145 return ESP_OK;
146 }
147 prev = curr;
148 curr = curr->next;
149 }
150
151 return ESP_ERR_NOT_FOUND;
152}

166static esp_err_t control_task_msg_dispatch(
167 dev_struct_t * pdev,
168 control_task_msg_code_t msg_code,
169 control_task_msg_evt_t evt,
170 void* params)
171{
172 if (!pdev)
173 return ESP_ERR_INVALID_ARG;
174
175 control_task_evt_cb_reg_t *ptr = control_task_msg_code_list_heads[msg_code];
176 bool evt_handled = false;
177
178 if (ptr == NULL) {
179 ESP_LOGW(TAG, "No control task msg callback registered for msg: %p", (void*)msg_code);
180 return ESP_ERR_INVALID_STATE;
181 }
182
183 ESP_LOGD(TAG, "msg|evt: %p|%p", (void*) msg_code, (void*) evt);
184
185 while (ptr) {
186 if ((evt & ptr->msg_evt_bmap) && (ptr->cb != NULL)) {
187 ptr->cb(pdev, evt, params); // Call the registered callback
188 evt_handled = true;
189 }
190 ptr = ptr->next; // Move to the next registration
191 }
192 if (!evt_handled)
193 ESP_LOGD(TAG, "No handler reg for EVT %p", (void*) evt);
194
195 return ESP_OK;
196}

224static void control_task_handler(void *args)
225{
226 esp_err_t err;
227 control_task_msg_t recv_msg;
228 dev_struct_t * pdev = (dev_struct_t *) args;
229 err = create_control_task_msg_q();
230 if (err)
231 ESP_LOGE(TAG, "Failed to initialise Control Task Msg Q Err: 0x%x", err);
232
233 while (true) {
234 if (xQueueReceive(control_task_queue, &recv_msg, portMAX_DELAY) == pdTRUE) {
235 err = control_task_msg_dispatch(pdev, recv_msg.msg_code, recv_msg.msg_evt, recv_msg.msg_evt_params);
236 if (err)
237 ESP_LOGE(TAG, "Err: 0x%x", err);
238 if (recv_msg.msg_evt_params) {
239 /* If Params were passed Free the allocated memory */
240 vPortFree(recv_msg.msg_evt_params);
241 ESP_LOGD(TAG, "ESP Heap available: %d", xPortGetFreeHeapSize());
242 }
243 }
244 }
