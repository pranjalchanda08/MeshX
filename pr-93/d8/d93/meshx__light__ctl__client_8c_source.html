41static meshx_err_t meshx_ctl_state_change_notify(meshx_gen_light_cli_cb_param_t *param)
42{
43 if (!param)
44 return MESHX_INVALID_ARG;
45
46 meshx_ctl_cli_el_msg_t el_light_ctl_param = {
47 .err_code = MESHX_SUCCESS,
48 .ctx = param->ctx,
49 .model = param->model
50 };
51 switch (param->ctx.opcode)
52 {
53 case MESHX_MODEL_OP_LIGHT_CTL_STATUS:
54 el_light_ctl_param.ctl_state.lightness = param->status.ctl_status.present_ctl_lightness;
55 el_light_ctl_param.ctl_state.temperature = param->status.ctl_status.present_ctl_temperature;
56 break;
57 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS:
58 el_light_ctl_param.ctl_state.delta_uv = param->status.ctl_temperature_status.present_ctl_delta_uv;
59 el_light_ctl_param.ctl_state.temperature = param->status.ctl_temperature_status.present_ctl_temperature;
60 break;
61 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS:
62 el_light_ctl_param.ctl_state.delta_uv = param->status.ctl_temperature_range_status.range_max;
63 el_light_ctl_param.ctl_state.temperature = param->status.ctl_temperature_range_status.range_min;
64 break;
65 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS:
66 el_light_ctl_param.ctl_state.delta_uv = param->status.ctl_default_status.delta_uv;
67 el_light_ctl_param.ctl_state.lightness = param->status.ctl_default_status.lightness;
68 el_light_ctl_param.ctl_state.temperature = param->status.ctl_default_status.temperature;
69 break;
70
71 default:
72 break;
73 }
74 if(param->evt == MESHX_GEN_LIGHT_CLI_TIMEOUT)
75 {
76 el_light_ctl_param.err_code = MESHX_TIMEOUT;
77 }
78
79 return control_task_msg_publish(
80 CONTROL_TASK_MSG_CODE_EL_STATE_CH,
81 CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_CTL,
82 &el_light_ctl_param,
83 sizeof(meshx_ctl_cli_el_msg_t));
84
85}

161static meshx_err_t meshx_handle_gen_light_msg(
162 const dev_struct_t *pdev,
163 control_task_msg_evt_t model_id,
164 meshx_gen_light_cli_cb_param_t *param
165)
166{
167 if (model_id != MESHX_MODEL_ID_LIGHT_CTL_CLI || !pdev || !param)
168 return MESHX_INVALID_ARG;
169
170 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "op|src|dst:%04" PRIx32 "|%04x|%04x",
171 param->ctx.opcode, param->ctx.src_addr, param->ctx.dst_addr);
172 meshx_err_t err = MESHX_SUCCESS;
173
174 switch (param->evt)
175 {
176 case MESHX_GEN_LIGHT_CLI_EVT_GET:
177 case MESHX_GEN_LIGHT_CLI_EVT_SET:
178 case MESHX_GEN_LIGHT_CLI_PUBLISH:
179 err = meshx_ctl_state_change_notify(param);
180 if (err)
181 {
182 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Failed to notify state change: %d", err);
183 }
184 break;
185 case MESHX_GEN_LIGHT_CLI_TIMEOUT:
186 err = meshx_light_ctl_client_timeout_handler();
187 if(err != MESHX_SUCCESS)
188 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Resend failed: %d", err);
189
190 else
191 err = meshx_ctl_state_change_notify(param);
192 break;
193 default:
194 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Unknown event: %d", param->evt);
195 err = MESHX_NOT_SUPPORTED;
196 break;
197 }
198
199 return err;
200}

341meshx_err_t meshx_light_ctl_client_send_msg(
342 meshx_light_ctl_client_model_t *model,
343 uint16_t opcode, uint16_t addr,
344 uint16_t net_idx, uint16_t app_idx,
345 uint16_t lightness, uint16_t temperature,
346 uint16_t delta_uv, uint8_t tid
347)
348{
349 meshx_err_t err;
350 meshx_light_client_set_state_t set = {0};
351 if (!model || !model->meshx_light_ctl_client_sig_model)
352 {
353 return MESHX_INVALID_ARG; // Invalid model pointer
354 }
355
356 light_ctl_client_last_msg_ctx.addr = addr;
357 light_ctl_client_last_msg_ctx.opcode = opcode;
358 light_ctl_client_last_msg_ctx.net_idx = net_idx;
359 light_ctl_client_last_msg_ctx.app_idx = app_idx;
360 light_ctl_client_last_msg_ctx.p_model = model;
361
362 if (opcode == MESHX_MODEL_OP_LIGHT_CTL_GET)
363 {
364 err = meshx_gen_light_send_msg(
365 model->meshx_light_ctl_client_sig_model,
366 &set, opcode,
367 addr, net_idx,
368 app_idx
369 );
370 }
371 else if (opcode == MESHX_MODEL_OP_LIGHT_CTL_SET ||
372 opcode == MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK)
373 {
374 set.ctl_set.tid = tid;
375 set.ctl_set.op_en = false;
376 set.ctl_set.ctl_delta_uv = delta_uv;
377 set.ctl_set.ctl_lightness = lightness;
378 set.ctl_set.ctl_temperature = temperature;
379
380 memcpy(&light_ctl_client_last_msg_ctx.state, &set, sizeof(meshx_light_client_set_state_t));
381
382 err = meshx_gen_light_send_msg(
383 model->meshx_light_ctl_client_sig_model,
384 &set, opcode,
385 addr, net_idx,
386 app_idx
387 );
388 }
389 else{
390 err = MESHX_INVALID_ARG; // Invalid opcode
391 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid opcode for Light CTL Client: %04x", opcode);
392 }
393 return err;
394}

538meshx_err_t meshx_light_ctl_state_change_handle(
539 meshx_ctl_cli_el_msg_t *param,
540 meshx_ctl_el_state_t *p_ctl_prev_state,
541 meshx_ctl_el_state_t *p_ctl_next_state
542)
543{
544 if (!p_ctl_prev_state || !param || !p_ctl_next_state)
545 return MESHX_INVALID_ARG;
546
547 /* Kept for future use */
548 MESHX_UNUSED(p_ctl_next_state);
549 bool state_change = false;
550 if(param->err_code == MESHX_SUCCESS)
551 {
552 if (param->ctx.opcode == MESHX_MODEL_OP_LIGHT_CTL_STATUS)
553 {
554 if (p_ctl_prev_state->lightness != param->ctl_state.lightness
555 || p_ctl_prev_state->temperature != param->ctl_state.temperature
556 )
557 {
558 p_ctl_prev_state->lightness = param->ctl_state.lightness;
559 p_ctl_prev_state->temperature = param->ctl_state.temperature;
560 state_change = true;
561 }
562 }
563 else if (param->ctx.opcode == MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS)
564 {
565 if(p_ctl_prev_state->delta_uv != param->ctl_state.delta_uv
566 || p_ctl_prev_state->temperature != param->ctl_state.temperature)
567 {
568 p_ctl_prev_state->delta_uv = param->ctl_state.delta_uv;
569 p_ctl_prev_state->temperature = param->ctl_state.temperature;
570 state_change = true;
571 }
572 }
573 else if (param->ctx.opcode == MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS)
574 {
575 if (p_ctl_prev_state->temp_range_max != param->ctl_state.temp_range_max
576 || p_ctl_prev_state->temp_range_min != param->ctl_state.temp_range_min)
577 {
578 p_ctl_prev_state->temp_range_max = param->ctl_state.temp_range_max;
579 p_ctl_prev_state->temp_range_min = param->ctl_state.temp_range_min;
580 state_change = true;
581 }
582 }
583 else if (param->ctx.opcode == MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS)
584 {
585 MESHX_DO_NOTHING;
586 }
587 else
588 {
589 /* Return as No CTL related OPCode were received */
590 MESHX_DO_NOTHING;
591 }
592 }
593 else
594 {
595 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "CWWW Client Element Message: Error (%d)", param->err_code);
596 /* Retry sending the failed packet done by MeshX Light CTL Layer. Do not notify App */
597 MESHX_DO_NOTHING;
598 }
599 return state_change ? MESHX_SUCCESS : MESHX_INVALID_STATE;
600}
