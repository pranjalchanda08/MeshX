145meshx_err_t meshx_nvs_open(uint16_t cid, uint16_t pid, uint32_t commit_timeout_ms)
146{
147 if (meshx_nvs_inst.init == MESHX_NVS_INIT_MAGIC)
148 {
149 return MESHX_INVALID_STATE;
150 }
151
152 meshx_err_t err;
153
154#ifndef CONFIG_BLE_MESH_SPECIFIC_PARTITION
155 err = nvs_open(
156 MESHX_NVS_NAMESPACE,
157 NVS_READWRITE,
158 &(meshx_nvs_inst.meshx_nvs_handle));
159#else
160 err = nvs_open_from_partition(
161 MESHX_NVS_PARTITION,
162 MESHX_NVS_NAMESPACE,
163 NVS_READWRITE,
164 &(meshx_nvs_inst.meshx_nvs_handle));
165#endif /* CONFIG_BLE_MESH_SPECIFIC_PARTITION */
166 if (err)
167 {
168 MESHX_LOGE(MODULE_ID_COMPONENT_MESHX_NVS, "nvs_open %p", (void *)err);
169 return err;
170 }
171
172#if MESHX_NVS_TIMER_PERIOD
173 if(commit_timeout_ms == 0)
174 {
175 commit_timeout_ms = MESHX_NVS_TIMER_PERIOD;
176 }
177 err = meshx_os_timer_create(
178 MESHX_NVS_TIMER_NAME,
179 commit_timeout_ms,
180 MESHX_NVS_RELOAD_ONE_SHOT,
181 &meshx_nvs_os_timer_cb,
182 &(meshx_nvs_inst.meshx_nvs_stability_timer));
183 if (err)
184 {
185 MESHX_LOGE(MODULE_ID_COMPONENT_MESHX_NVS, "os_timer_create %p", (void *)err);
186 return err;
187 }
188#else
189 MESHX_UNUSED(commit_timeout_ms);
190#endif /* MESHX_NVS_TIMER_PERIOD */
191
192 meshx_nvs_inst.init = MESHX_NVS_INIT_MAGIC;
193
194 err = meshx_nvs_get(
195 MESHX_NVS_NAMESPACE_CID,
196 &(meshx_nvs_inst.cid),
197 sizeof(meshx_nvs_inst.cid));
198
199 err += meshx_nvs_get(
200 MESHX_NVS_NAMESPACE_PID,
201 &(meshx_nvs_inst.pid),
202 sizeof(meshx_nvs_inst.pid));
203
204 if(err != MESHX_SUCCESS)
205 {
206 MESHX_LOGW(MODULE_ID_COMPONENT_MESHX_NVS, "Product ID not found in NVS reinitializing MeshX NVS");
207 err = meshx_nvs_erase_prod_init(cid, pid);
208 }
209
210 else
211 {
212 if (meshx_nvs_inst.cid == cid && meshx_nvs_inst.pid == pid)
213 {
214 MESHX_LOGI(MODULE_ID_COMPONENT_MESHX_NVS, "Product ID match: %x|%x", meshx_nvs_inst.pid, meshx_nvs_inst.cid);
215 }
216 else
217 {
218 MESHX_LOGW(MODULE_ID_COMPONENT_MESHX_NVS, "Product ID mismatch: %x|%x", meshx_nvs_inst.pid, meshx_nvs_inst.cid);
219 err = meshx_nvs_erase_prod_init(cid, pid);
220 }
221 }
222
223 return err;
224}

437static meshx_err_t meshx_nvs_unit_test_cb_handler(int cmd_id, int argc, char **argv)
438{
439 meshx_err_t err = MESHX_SUCCESS;
440 uint32_t ut_blob = 0xDEAD;
441 uint32_t ut_blob_get = 0x00;
442 bool arm_timer;
443
444 MESHX_UNUSED(argc);
445 MESHX_LOGD(MODULE_ID_COMPONENT_MESHX_NVS, "argc|cmd_id: %d|%d", argc, cmd_id);
446 if (cmd_id >= MESHX_NVS_CLI_MAX)
447 {
448 MESHX_LOGE(MODULE_ID_COMPONENT_MESHX_NVS, "Invalid number of arguments");
449 return MESHX_INVALID_ARG;
450 }
451
452 switch(cmd_id)
453 {
454 case MESHX_NVS_CLI_CMD_OPEN:
455 /* MeshX NVS Open uses defaut values */
456 err = meshx_nvs_open(CONFIG_CID_ID, CONFIG_PID_ID, 0);
457 break;
458 case MESHX_NVS_CLI_CMD_SET:
459 arm_timer = UT_GET_ARG(0, bool, argv) == 0 ? MESHX_NVS_NO_AUTO_COMMIT : MESHX_NVS_AUTO_COMMIT;
460 err = meshx_nvs_set(MESHX_NVS_UNIT_TEST_KEY, &ut_blob, sizeof(ut_blob), arm_timer);
461 break;
462 case MESHX_NVS_CLI_CMD_GET:
463 err = meshx_nvs_get(MESHX_NVS_UNIT_TEST_KEY, &ut_blob_get, sizeof(ut_blob_get));
464 if(err == MESHX_SUCCESS && ut_blob != ut_blob_get)
465 MESHX_LOGE(MODULE_ID_COMPONENT_MESHX_NVS, "MESHX NVS Integrety Test Failed");
466 break;
467 case MESHX_NVS_CLI_CMD_COMMIT:
468 err = meshx_nvs_commit();
469 break;
470 case MESHX_NVS_CLI_CMD_REMOVE:
471 err = meshx_nvs_remove(MESHX_NVS_UNIT_TEST_KEY);
472 break;
473 case MESHX_NVS_CLI_CMD_ERASE:
474 err = meshx_nvs_erase();
475 break;
476 case MESHX_NVS_CLI_CMD_CLOSE:
477 err = meshx_nvs_close();
478 break;
479 default:
480 err = MESHX_INVALID_ARG;
481 break;
482 }
483
484 return err;
485}
