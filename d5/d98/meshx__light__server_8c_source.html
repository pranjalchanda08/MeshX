33meshx_err_t meshx_is_status_in_gen_light_grp(uint16_t opcode)
34{
35 switch (opcode)
36 {
37 case MESHX_MODEL_OP_LIGHT_LIGHTNESS_STATUS:
38 case MESHX_MODEL_OP_LIGHT_LIGHTNESS_LINEAR_STATUS:
39 case MESHX_MODEL_OP_LIGHT_CTL_STATUS:
40 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS:
41 case MESHX_MODEL_OP_LIGHT_HSL_STATUS:
42 case MESHX_MODEL_OP_LIGHT_HSL_HUE_STATUS:
43 case MESHX_MODEL_OP_LIGHT_HSL_SATURATION_STATUS:
44 return MESHX_SUCCESS;
45 default:
46 return MESHX_FAIL;
47 }
48}
