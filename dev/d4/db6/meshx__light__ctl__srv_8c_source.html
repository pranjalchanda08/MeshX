31static meshx_err_t meshx_state_change_notify(meshx_lighting_server_cb_param_t *param)
32{
33 meshx_light_ctl_srv_t ctl_srv_params = {
34 .model = param->model};
35
36 if (MESHX_ADDR_IS_UNICAST(param->ctx.dst_addr) || (MESHX_ADDR_BROADCAST(param->ctx.dst_addr)) || (MESHX_ADDR_IS_GROUP(param->ctx.dst_addr) && (MESHX_SUCCESS == meshx_is_group_subscribed(param->model.p_model, param->ctx.dst_addr))))
37 {
38 switch (param->ctx.opcode)
39 {
40 case MESHX_MODEL_OP_LIGHT_CTL_SET:
41 case MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK:
42 ctl_srv_params.state.delta_uv = param->state_change.ctl_set.delta_uv;
43 ctl_srv_params.state.lightness = param->state_change.ctl_set.lightness;
44 ctl_srv_params.state.temperature = param->state_change.ctl_set.temperature;
45 break;
46 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET:
47 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET:
48 ctl_srv_params.state.delta_uv = param->state_change.ctl_temp_set.delta_uv;
49 ctl_srv_params.state.temperature = param->state_change.ctl_temp_set.temperature;
50 break;
51 default:
52 return MESHX_NOT_SUPPORTED;
53 }
54 /* Send msg for hw manipulation */
55 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "HW change requested, Element_id: 0x%x",
56 param->model.el_id);
57
58 meshx_err_t err = control_task_msg_publish(
59 CONTROL_TASK_MSG_CODE_EL_STATE_CH,
60 CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_CTL,
61 &ctl_srv_params,
62 sizeof(meshx_light_ctl_srv_t));
63 return err;
64 }
65 return MESHX_NOT_SUPPORTED;
66}

82static meshx_err_t meshx_handle_light_ctl_msg(esp_ble_mesh_lighting_server_cb_param_t *param)

100static meshx_err_t meshx_handle_light_ctl_msg(const dev_struct_t *pdev,
101 const control_task_msg_evt_t evt,
102 meshx_lighting_server_cb_param_t *param)
103{
104 if (!pdev || (evt != MESHX_MODEL_ID_LIGHT_CTL_SRV && evt != MESHX_MODEL_ID_LIGHT_CTL_SETUP_SRV))
105 return MESHX_INVALID_ARG;
106#endif /* !CONFIG_BLE_CONTROL_TASK_OFFLOAD_ENABLE */
107 meshx_err_t err = MESHX_SUCCESS;
108
109 bool send_reply_to_src = false;
110 bool state_change_notify = false;
111 uint32_t op_code = param->ctx.opcode;
112 uint32_t status_op = 0;
113 switch (op_code)
114 {
115 case MESHX_MODEL_OP_LIGHT_CTL_STATUS:
116 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS:
117 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS:
118 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS:
119 /* Ignore, as these are status messages */
120 break;
122 case MESHX_MODEL_OP_LIGHT_CTL_GET:
123 case MESHX_MODEL_OP_LIGHT_CTL_SET:
124 case MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK:
125 status_op = MESHX_MODEL_OP_LIGHT_CTL_STATUS;
126 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_GET)
127 state_change_notify = true;
128
129 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK)
130 send_reply_to_src = true;
131
132 break;
133 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET:
134 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET:
135 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET_UNACK:
136 status_op = MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS;
137 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET)
138 state_change_notify = true;
139
140 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET_UNACK)
141 send_reply_to_src = true;
142 break;
144 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET:
145 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_GET:
146 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET_UNACK:
147 status_op = MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS;
148 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_GET)
149 MESHX_DO_NOTHING;
150
151 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET_UNACK)
152 send_reply_to_src = true;
153
154 break;
155 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET:
156 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET:
157 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK:
158 status_op = MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS;
159 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET)
160 MESHX_DO_NOTHING;
161
162 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK)
163 send_reply_to_src = true;
164
165 break;
166 default:
167 MESHX_LOGW(MODULE_ID_MODEL_SERVER, "CTL Unhandled Event %p", (void *)param->ctx.opcode);
168 break;
169 }
170 if (state_change_notify)
171 {
172 err = meshx_state_change_notify(param);
173 if (err)
174 return err;
175 }
176 if (send_reply_to_src
177 /* This is meant to notify the respective publish client */
178 || param->ctx.src_addr != param->model.pub_addr)
179 {
180 /* Here the message was received from unregistered source and mention the state to the respective client */
181 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "PUB: src|pub %x|%x", param->ctx.src_addr, param->model.pub_addr);
182 param->ctx.opcode = (uint16_t)status_op;
183 param->ctx.dst_addr = param->model.pub_addr;
184
185 err = meshx_gen_light_srv_send_msg_to_ble(
186 CONTROL_TASK_MSG_EVT_TO_BLE_SET_CTL_SRV,
187 param
188 );
189 }
190
191 return err;
192}
