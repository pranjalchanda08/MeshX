95static meshx_err_t meshx_prov_srv_control_task_handler(
96 dev_struct_t *pdev,
97 control_task_msg_evt_t evt,
98 meshx_prov_srv_param_t *params)
99{
100 if (!pdev || !params)
101 {
102 return MESHX_INVALID_ARG;
103 }
104 prov_evt_t prov_evt = CONTROL_TASK_MSG_EVT_PROVISION_ALL;
105 if(evt != CONTROL_TASK_MSG_EVT_PROVISION_ALL)
106 {
107 return MESHX_INVALID_ARG;
108 }
109
110 if (prov_cb_evt_ctrl_task_evt_table[params->prov_evt].ctrl_task_evt != 0)
111 {
112 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "Provisioning event mapped: %s",
113 prov_cb_evt_ctrl_task_evt_table[params->prov_evt].evt_str);
114 prov_evt = prov_cb_evt_ctrl_task_evt_table[params->prov_evt].ctrl_task_evt;
115 }
116 else
117 {
118 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "Unhandled event: %d", event);
119 }
120 if (params->prov_evt == MESHX_NODE_PROV_COMPLETE_EVT)
121 {
122 MESHX_LOGI(MODULE_ID_MODEL_SERVER, "net_idx: 0x%04x, addr: 0x%04x", params->param.node_prov_complete.net_idx, params->param.node_prov_complete.addr);
123 MESHX_LOGI(MODULE_ID_MODEL_SERVER, "flags: 0x%02x, iv_index: 0x%08" PRIx32, params->param.node_prov_complete.flags, params->param.node_prov_complete.iv_index);
124 }
125 if (prov_evt == CONTROL_TASK_MSG_EVT_PROVISION_ALL)
126 {
127 return MESHX_INVALID_ARG;
128 }
129
130 /* Publish the event */
131 return control_task_msg_publish(
132 CONTROL_TASK_MSG_CODE_PROVISION,
133 prov_evt,
134 &params->param,
135 sizeof(meshx_prov_cb_param_t));
136 return MESHX_SUCCESS;
137}

148static void meshx_handle_node_reset(dev_struct_t *pdev)
149{
150 if(!pdev)
151 {
152 MESHX_LOGE(MODULE_ID_COMMON, "Invalid device structure");
153 return;
154 }
155
156 for(uint16_t i = 1; i < pdev->element_idx; i++)
157 {
158 meshx_err_t err = meshx_nvs_element_ctx_remove(i);
159 if(err != MESHX_SUCCESS)
160 {
161 MESHX_LOGE(MODULE_ID_COMMON, "Failed to erase element context (%d): (%d)", i, err);
162 }
163 }
164 /* Reset the MCU */
165 meshx_platform_reset();
166}

181static meshx_err_t meshx_prov_control_task_handler(dev_struct_t *pdev, control_task_msg_evt_t evt, void *params)
182{
183 const meshx_prov_cb_param_t *param = (meshx_prov_cb_param_t*) params;
184
185 switch (evt)
186 {
187 case CONTROL_TASK_MSG_EVT_PROVISION_STOP:
188 pdev->meshx_store.net_key_id = param->node_prov_complete.net_idx;
189 pdev->meshx_store.node_addr = param->node_prov_complete.addr;
190 meshx_nvs_set(MESHX_NVS_STORE, &pdev->meshx_store, sizeof(pdev->meshx_store), MESHX_NVS_AUTO_COMMIT);
191 break;
192 case CONTROL_TASK_MSG_EVT_IDENTIFY_START:
193 MESHX_LOGI(MODULE_ID_COMMON, "Identify Start");
194 break;
195 case CONTROL_TASK_MSG_EVT_NODE_RESET:
196 MESHX_LOGW(MODULE_ID_COMMON, "Node Reset Event");
197 meshx_handle_node_reset(pdev);
198 break;
199 default:
200 break;
201 }
202 return MESHX_SUCCESS;
203}

254static meshx_err_t meshx_init_freshboot_timer(dev_struct_t *p_dev, uint16_t timeout_ms)
255{
256 if(p_dev->meshx_store.node_addr == MESHX_ADDR_UNASSIGNED)
257 {
258 MESHX_LOGI(MODULE_ID_COMMON, "Device not provisioned, not starting boot timer");
259 return MESHX_SUCCESS;
260 }
261 meshx_err_t err = meshx_os_timer_create("boot_timer",
262 timeout_ms,
263 false,
264 meshx_init_freshboot_timer_trigger_cb,
265 &g_boot_timer
266 );
267 if(err != MESHX_SUCCESS)
268 {
269 MESHX_LOGE(MODULE_ID_COMMON, "Failed to create boot timer: (%d)", err);
270 }
271
272 err = meshx_os_timer_start(g_boot_timer);
273 if(err != MESHX_SUCCESS)
274 {
275 MESHX_LOGE(MODULE_ID_COMMON, "Failed to start boot timer: (%d)", err);
276 }
277
278 return err;
279}

313meshx_err_t meshx_init_prov(dev_struct_t *p_dev, const meshx_prov_params_t *prov_cfg)
314{
315 if (!prov_cfg || memcmp(prov_cfg->uuid, MESHX_UUID_EMPTY, sizeof(meshx_uuid_addr_t)) == 0)
316 {
317 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Invalid server configuration");
318 return MESHX_INVALID_ARG;
319 }
320
321 meshx_err_t err = meshx_prov_srv_reg_from_ble_cb();
322 if (err != MESHX_SUCCESS)
323 {
324 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to register provisioning callback");
325 return err;
326 }
327
328 err = meshx_prov_srv_meshx_reg_cb();
329 if (err != MESHX_SUCCESS)
330 {
331 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to register provisioning callback");
332 return err;
333 }
334
335 err = meshx_init_freshboot_timer(p_dev, prov_cfg->freshboot_timeout_ms);
336 if (err != MESHX_SUCCESS)
337 {
338 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to initialize boot timer");
339 return err;
340 }
341
342 return meshx_plat_init_prov(prov_cfg->uuid);
343}
