114MESHX_MODEL * get_root_sig_models(void)

116 static MESHX_MODEL temp_model;
117 if(meshx_sig_root_model_arr == NULL && meshx_sig_root_model_arr_len)
118 {
119 meshx_sig_root_model_arr = (MESHX_MODEL *) MESHX_MALLOC(sizeof(MESHX_MODEL) * meshx_sig_root_model_arr_len);

201meshx_err_t meshx_create_element_composition(dev_struct_t *p_dev, meshx_config_t const *config)
202{
203#if CONFIG_MAX_ELEMENT_COUNT > 0
204 meshx_err_t err;
205#if CONFIG_SECTION_ENABLE_ELEMENT_TABLE
206 element_comp_table_t *element_comp_table;
207#endif /* CONFIG_SECTION_ENABLE_ELEMENT_TABLE */
208 if(!p_dev || !config || !config->element_comp_arr_len || !config->element_comp_arr)
209 return MESHX_INVALID_ARG;
210
211 err = meshx_init_config_server();
212 MESHX_ERR_PRINT_RET("Failed to initialize config server", err);
213 for(uint16_t element_id = 0; element_id < config->element_comp_arr_len; element_id++)
214 {
215#if !CONFIG_SECTION_ENABLE_ELEMENT_TABLE
216 if(config->element_comp_arr[element_id].element_cnt != 0
217 && element_comp_fn[config->element_comp_arr[element_id].type] != NULL)
218 {
219 err = element_comp_fn[config->element_comp_arr[element_id].type](p_dev, config->element_comp_arr[element_id].element_cnt);
220 if(err)
221 {
222 MESHX_LOGE(MODULE_ID_COMMON, "Element composition failed: (%d)", err);
223 return err;
224 }
225 }
226#else
227 element_comp_table = &MESHX_ELEMENT_COMP_TABLE_START;
228 while(element_comp_table < &MESHX_ELEMENT_COMP_TABLE_STOP)
229 {
230 if(element_comp_table->idx == element_id && config->element_comp_arr[element_id].element_cnt != 0)
231 {
232 err = element_comp_table->element_comp_fn(p_dev, config->element_comp_arr[element_id].element_cnt);
233 if(err)
234 {
235 MESHX_LOGE(MODULE_ID_COMMON, "Element composition failed: (%d)", err);
236 return err;
237 }
238 }
239 element_comp_table++;
240 }
241#endif /* CONFIG_SECTION_ENABLE_ELEMENT_TABLE */
242 }
243#endif /* CONFIG_MAX_ELEMENT_COUNT */
244 return err;
245}
