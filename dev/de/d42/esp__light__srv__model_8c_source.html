66static void meshx_ble_lightness_server_cb(MESHX_LIGHT_SRV_CB_EVT event,
67 MESHX_LIGHT_SRV_CB_PARAM *param)
68{
69 MESHX_UNUSED(event);
70 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "evt|op|src|dst: %02x|%04x|%04x|%04x|%04x",
71 event, (unsigned)param->ctx.recv_op, param->ctx.addr, param->ctx.recv_dst,
72 param->model->model_id);
73
74 if(event != ESP_BLE_MESH_LIGHTING_SERVER_STATE_CHANGE_EVT)
75 {
76 return;
77 }
78 MESHX_LIGHT_CTL_SRV *srv = (MESHX_LIGHT_CTL_SRV *)param->model->user_data;
79
80 bool publish_flag = false;
81 uint32_t op_code = param->ctx.recv_op;
82 meshx_lighting_server_cb_param_t pub_param = {
83 .ctx = {
84 .net_idx = param->ctx.net_idx,
85 .app_idx = param->ctx.app_idx,
86 .dst_addr = param->ctx.recv_dst,
87 .src_addr = param->ctx.addr,
88 .opcode = param->ctx.recv_op,
89 .p_ctx = &param->ctx
90 },
91 .model = {
92 .el_id = param->model->element_idx,
93 .pub_addr = param->model->pub->publish_addr,
94 .model_id = param->model->model_id,
95 .p_model = param->model
96 }
97 };
98
99 switch (op_code)
100 {
102 case MESHX_MODEL_OP_LIGHT_CTL_GET:
103 case MESHX_MODEL_OP_LIGHT_CTL_SET:
104 case MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK:
105 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_GET)
106 {
107 srv->state->temperature = param->value.state_change.ctl_set.temperature;
108 srv->state->lightness = param->value.state_change.ctl_set.lightness;
109 srv->state->delta_uv = param->value.state_change.ctl_set.delta_uv;
110 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "lightness|temp|del_uv:%d|%d|%d",
111 srv->state->lightness,
112 srv->state->temperature,
113 srv->state->delta_uv);
114
115 pub_param.state_change.ctl_set.delta_uv = srv->state->delta_uv;
116 pub_param.state_change.ctl_set.lightness = srv->state->lightness;
117 pub_param.state_change.ctl_set.temperature = srv->state->temperature;
118
119 publish_flag = true;
120 }
121 break;
122 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET:
123 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET:
124 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET_UNACK:
125 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET)
126 {
127 srv->state->temperature = param->value.state_change.ctl_temp_set.temperature;
128 srv->state->delta_uv = param->value.state_change.ctl_temp_set.delta_uv;
129 MESHX_LOGI(MODULE_ID_MODEL_SERVER, "lightness|del_uv:%d|%d",
130 srv->state->temperature,
131 srv->state->delta_uv);
132
133 pub_param.state_change.ctl_temp_set.delta_uv = srv->state->delta_uv;
134 pub_param.state_change.ctl_temp_set.temperature = srv->state->temperature;
135
136 publish_flag = true;
137 }
138 break;
140 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET:
141 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_GET:
142 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET_UNACK:
143 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_GET)
144 {
145 MESHX_LOGI(MODULE_ID_MODEL_SERVER, "lightness|temp|del_uv:%d|%d|%d",
146 param->value.state_change.ctl_default_set.lightness,
147 param->value.state_change.ctl_default_set.temperature,
148 param->value.state_change.ctl_default_set.delta_uv);
149 srv->state->temperature_default = param->value.state_change.ctl_default_set.temperature;
150 srv->state->lightness_default = param->value.state_change.ctl_default_set.lightness;
151 srv->state->delta_uv_default = param->value.state_change.ctl_default_set.delta_uv;
152
153 pub_param.state_change.ctl_default_set.delta_uv = srv->state->delta_uv_default;
154 pub_param.state_change.ctl_default_set.lightness = srv->state->lightness_default;
155 pub_param.state_change.ctl_default_set.temperature = srv->state->temperature_default;
156
157 publish_flag = true;
158 }
159 break;
160 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET:
161 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET:
162 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK:
163 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET)
164 {
165 MESHX_LOGI(MODULE_ID_MODEL_SERVER, "temp min|max: %dK|%dK",
166 param->value.state_change.ctl_temp_range_set.range_min,
167 param->value.state_change.ctl_temp_range_set.range_max);
168 srv->state->temperature_range_min = param->value.state_change.ctl_temp_range_set.range_min;
169 srv->state->temperature_range_max = param->value.state_change.ctl_temp_range_set.range_max;
170
171 pub_param.state_change.ctl_temp_range_set.range_max = srv->state->temperature_range_max;
172 pub_param.state_change.ctl_temp_range_set.range_min = srv->state->temperature_range_min;
173
174 publish_flag = true;
175 }
176 break;
177 default:
178 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "CTL Unhandled Event %p", (meshx_ptr_t )param->ctx.recv_op);
179 break;
180 }
181 if (publish_flag)
182 control_task_msg_publish(CONTROL_TASK_MSG_CODE_FRM_BLE,
183 pub_param.model.model_id,
184 &pub_param,
185 sizeof(meshx_lighting_server_cb_param_t));
186}

188meshx_err_t meshx_plat_gen_light_srv_send_status(
189 const meshx_model_t *p_model,
190 const meshx_ctx_t *p_ctx,
191 const meshx_lighting_server_state_change_t *state_change)
192{
193 if (!p_model || !p_ctx || !state_change)
194 {
195 return MESHX_INVALID_ARG;
196 }
197 meshx_err_t err = MESHX_SUCCESS;
198 static esp_ble_mesh_msg_ctx_t ctx;
199 const esp_ble_mesh_msg_ctx_t *pctx = (esp_ble_mesh_msg_ctx_t *)p_ctx->p_ctx;
200 if(pctx != NULL)
201 {
202 memcpy(&ctx, pctx, sizeof(esp_ble_mesh_msg_ctx_t));
203 }
204 ctl_status_t ctl_status_union;
205 uint8_t ctl_status_pack_len = 0;
206
207 switch (p_ctx->opcode)
208 {
209 case MESHX_MODEL_OP_LIGHT_CTL_STATUS:
210 ctl_status_union.ctl_status.temperature = state_change->ctl_set.temperature;
211 ctl_status_union.ctl_status.lightness = state_change->ctl_set.lightness;
212 ctl_status_pack_len = sizeof(ctl_status_union.ctl_status);
213 break;
214 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS:
215 ctl_status_union.ctl_temp_status.temperature = state_change->ctl_temp_set.temperature;
216 ctl_status_union.ctl_temp_status.delta_uv = state_change->ctl_temp_set.delta_uv;
217 ctl_status_pack_len = sizeof(ctl_status_union.ctl_temp_status);
218 break;
219 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS:
220 ctl_status_union.ctl_default.delta_uv_def = state_change->ctl_default_set.delta_uv;
221 ctl_status_union.ctl_default.lightness_def = state_change->ctl_default_set.lightness;
222 ctl_status_union.ctl_default.temperature_def = state_change->ctl_default_set.temperature;
223 ctl_status_pack_len = sizeof(ctl_status_union.ctl_default);
224 break;
225 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS:
226 ctl_status_union.ctl_temp_range.status_code = MESHX_SUCCESS;
227 ctl_status_union.ctl_temp_range.range_min = state_change->ctl_temp_range_set.range_min;
228 ctl_status_union.ctl_temp_range.range_max = state_change->ctl_temp_range_set.range_max;
229 ctl_status_pack_len = sizeof(ctl_status_union.ctl_temp_range);
230 break;
231 default:
232 err = MESHX_INVALID_ARG;
233 }
234
235 ctx.net_idx = p_ctx->net_idx;
236 ctx.app_idx = p_ctx->app_idx;
237 ctx.addr = p_ctx->dst_addr;
238 ctx.send_ttl = ESP_BLE_MESH_TTL_DEFAULT;
239 ctx.send_cred = 0;
240 ctx.send_tag = BIT1;
241
242 esp_err_t esp_err = esp_ble_mesh_server_model_send_msg(p_model->p_model,
243 &ctx,
244 p_ctx->opcode,
245 ctl_status_pack_len,
246 (uint8_t *)&ctl_status_union);
247 if (esp_err)
248 {
249 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Mesh Model msg send failed (err: 0x%x)", esp_err);
250 return MESHX_ERR_PLAT;
251 }
252 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "Mesh Model msg sent (opcode: 0x%04x, len: %d)", p_ctx->opcode, ctl_status_pack_len);
253
254 return err;
255}
