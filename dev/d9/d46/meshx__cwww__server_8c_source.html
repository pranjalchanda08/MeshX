254static meshx_err_t meshx_restore_model_states(uint16_t element_id)
255{
256 uint16_t model_id = 0;

258 meshx_cwww_server_ctx_t const *el_ctx = cwww_element_init_ctrl.el_list[element_id].srv_ctx;
259
260 for (size_t i = 0; i < CWWW_SRV_MODEL_SIG_CNT; i++)
261 {
262 err = meshx_get_model_id(cwww_element_init_ctrl.el_list[element_id].onoff_srv_model->meshx_server_sig_model,
263 &model_id);
264 if (err)
265 {
266 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to get model ID (err: 0x%x)", err);
267 return err;
268 }
269 if (model_id == MESHX_MODEL_ID_GEN_ONOFF_SRV)
270 {
271 err = meshx_gen_on_off_srv_state_restore(cwww_element_init_ctrl.el_list[element_id].onoff_srv_model->meshx_server_sig_model,
272 el_ctx->prev_state);
273 if (err)
274 {
275 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to restore on-off server state (err: 0x%x)", err);
276 return err;
277 }
278 }
279 else if (model_id == MESHX_MODEL_ID_LIGHT_CTL_SRV)
280 {
281 err = meshx_light_ctl_srv_state_restore(cwww_element_init_ctrl.el_list[element_id].ctl_srv_model,
282 el_ctx->prev_ctl_state);
283 if (err)
284 {
285 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to restore on-off server state (err: 0x%x)", err);
286 return err;
287 }
288 }
289 }
290 return err;

302static meshx_err_t meshx_add_cwww_srv_model_to_element_list(dev_struct_t *pdev, uint16_t *start_idx, uint16_t n_max)
303{
304 if (!pdev)
305 return MESHX_INVALID_STATE;
306
307 if (!start_idx || (n_max + *start_idx) > CONFIG_MAX_ELEMENT_COUNT)
308 {
309 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "No of elements limit reached");
310 return MESHX_NO_MEM;
311 }
312 meshx_err_t err = MESHX_SUCCESS;
313 cwww_element_init_ctrl.element_id_start = *start_idx;
314
315 for (uint16_t i = *start_idx; i < (n_max + *start_idx); i++)
316 {

319
320 err = meshx_plat_add_element_to_composition(

322 pdev->elements,
323 cwww_element_init_ctrl.el_list[i - *start_idx].cwww_srv_model_list,
324 NULL,
325 CWWW_SRV_MODEL_SIG_CNT,
326 CWWW_SRV_MODEL_VEN_CNT);
327 if (err)
328 {

330 return err;
331 }
332
333 err = meshx_nvs_element_ctx_get(
334 i,
335 cwww_element_init_ctrl.el_list[i - *start_idx].srv_ctx,
336 sizeof(meshx_cwww_server_ctx_t));
337 if (err != MESHX_SUCCESS)
338 {
339 MESHX_LOGW(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to get cwww element context: (0x%x)", err);
340 }
341 else
342 {
343 err = meshx_restore_model_states(i - *start_idx);
344 if (err != MESHX_SUCCESS)
345 {
346 MESHX_LOGW(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to restore cwww model states: (0x%x)", err);
347 }
348 }
349 }
350 /* Increment the index for further registrations */
351 cwww_element_init_ctrl.element_id_end = *start_idx += n_max;
352 return MESHX_SUCCESS;
353}

375static meshx_err_t meshx_api_control_task_handler(const dev_struct_t *pdev, const control_task_msg_evt_t evt, const void *params)
376{
377 MESHX_UNUSED(pdev);
378 meshx_err_t err = MESHX_SUCCESS;
379 uint16_t element_id = 0;
380 size_t rel_el_id;
381 meshx_cwww_server_ctx_t *el_ctx = NULL;
382 meshx_api_light_cwww_server_evt_t app_msg;
383 cwww_sig_id_t sig_func = CWWW_SIG_ONOFF_MODEL_ID;
384
385 switch (evt)
386 {
387 case CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_ON_OFF:
388 {
389 const meshx_on_off_srv_el_msg_t *p_onoff_srv = (const meshx_on_off_srv_el_msg_t *)params;
390 element_id = p_onoff_srv->model.el_id;
391 if (!IS_EL_IN_RANGE(element_id))
392 goto el_ctrl_task_hndlr_exit;
393
394 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
395 el_ctx = cwww_element_init_ctrl.el_list[rel_el_id].srv_ctx;
396 if (el_ctx->prev_state.on_off == p_onoff_srv->on_off_state)
397 goto el_ctrl_task_hndlr_exit;
398
399 sig_func = CWWW_SIG_ONOFF_MODEL_ID;
400 el_ctx->prev_state.on_off = p_onoff_srv->on_off_state;
401 app_msg.state_change.on_off.state = el_ctx->prev_state.on_off;
402 break;
403 }
404
405 case CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_CTL:
406 {
407 const meshx_light_ctl_srv_t *p_ctl_srv = (const meshx_light_ctl_srv_t *)params;
408 element_id = p_ctl_srv->model.el_id;
409 if (!IS_EL_IN_RANGE(element_id))
410 goto el_ctrl_task_hndlr_exit;
411
412 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
413 el_ctx = cwww_element_init_ctrl.el_list[rel_el_id].srv_ctx;
414
415 if (el_ctx->prev_ctl_state.delta_uv == p_ctl_srv->state.delta_uv &&
416 el_ctx->prev_ctl_state.lightness == p_ctl_srv->state.lightness &&
417 el_ctx->prev_ctl_state.temperature == p_ctl_srv->state.temperature &&
418 el_ctx->prev_ctl_state.temperature_range_min == p_ctl_srv->state.temperature_range_min &&
419 el_ctx->prev_ctl_state.temperature_range_max == p_ctl_srv->state.temperature_range_max)
420 goto el_ctrl_task_hndlr_exit;
421
422 el_ctx->prev_ctl_state.delta_uv = p_ctl_srv->state.delta_uv;
423 el_ctx->prev_ctl_state.lightness = p_ctl_srv->state.lightness;
424 el_ctx->prev_ctl_state.temperature = p_ctl_srv->state.temperature;
425 el_ctx->prev_ctl_state.temperature_range_min = p_ctl_srv->state.temperature_range_min;
426 el_ctx->prev_ctl_state.temperature_range_max = p_ctl_srv->state.temperature_range_max;
427
428 sig_func = CWWW_SIG_L_CTL_MODEL_ID;
429 app_msg.state_change.ctl.delta_uv = el_ctx->prev_ctl_state.delta_uv;
430 app_msg.state_change.ctl.lightness = el_ctx->prev_ctl_state.lightness;
431 app_msg.state_change.ctl.temperature = el_ctx->prev_ctl_state.temperature;
432 app_msg.state_change.ctl.temp_range_min = el_ctx->prev_ctl_state.temperature_range_min;
433 app_msg.state_change.ctl.temp_range_max = el_ctx->prev_ctl_state.temperature_range_max;
434 break;
435 }
436 default:
437 break;
438 }
439
440 err = meshx_nvs_element_ctx_set(element_id, el_ctx, sizeof(meshx_cwww_server_ctx_t));
441 if (err != MESHX_SUCCESS)
442 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to set relay element context: (%d)", err);
443
444 err = meshx_send_msg_to_app(element_id,
445 MESHX_ELEMENT_TYPE_LIGHT_CWWW_SERVER,
446 (uint16_t)sig_func,
447 sizeof(meshx_api_light_cwww_server_evt_t),
448 &app_msg);
449
450 if (err != MESHX_SUCCESS)
451 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to send relay state change message: (%d)", err);
452
453el_ctrl_task_hndlr_exit:
454 return MESHX_SUCCESS;
455}

539static meshx_err_t meshx_cwww_srv_msg_send_handler(
540 const dev_struct_t *pdev,
541 control_task_msg_evt_to_ble_t evt,
542 void *params)
543{
544 MESHX_UNUSED(pdev);
545 if((evt & (CONTROL_TASK_MSG_EVT_TO_BLE_GEN_SRV_MASK)) == 0)
546 return MESHX_SUCCESS;
547 uint16_t element_id = 0;

551 switch (evt)
552 {
553 case CONTROL_TASK_MSG_EVT_TO_BLE_SET_ON_OFF_SRV:
554 {
555 gen_srv_send = (meshx_gen_srv_cb_param_t *)params;

560 &gen_srv_send->model,
561 &gen_srv_send->ctx,
562 gen_srv_send->state_change.onoff_set.onoff
563 );

565 {
566 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to send ONOFF status message (Err: %x)", err);

568 }
569 }
570 break;
571 case CONTROL_TASK_MSG_EVT_TO_BLE_SET_CTL_SRV:
572 {
573 light_srv_send = (meshx_lighting_server_cb_param_t *)params;
574 element_id = light_srv_send->model.el_id;

578 &light_srv_send->model,
579 &light_srv_send->ctx,
580 light_srv_send->state_change.ctl_set.delta_uv,

583 );

586 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to send CTL status message (Err: %x)", err);

588 }
