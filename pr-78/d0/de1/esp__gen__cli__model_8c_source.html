71static void esp_ble_mesh_generic_client_cb(MESHX_GEN_CLI_CB_EVT event,
72 MESHX_GEN_CLI_CB_PARAM *param)
73{
74 MESHX_UNUSED(client_state_str);
75 MESHX_LOGD(MODULE_ID_MODEL_CLIENT, "%s, err|op|src|dst: %d|%04" PRIx32 "|%04x|%04x",
76 client_state_str[event], param->error_code,
77 param->params->ctx.recv_op, param->params->ctx.addr, param->params->ctx.recv_dst);
78
79 meshx_gen_cli_cb_param_t pub_param = {
80 .ctx = {
81 .net_idx = param->params->ctx.net_idx,
82 .app_idx = param->params->ctx.app_idx,
83 .dst_addr = param->params->ctx.recv_dst,
84 .src_addr = param->params->ctx.addr,
85 .opcode = param->params->ctx.recv_op,
86 .p_ctx = &param->params->ctx
87 },
88 .model = {
89 .pub_addr = param->params->model->pub->publish_addr,
90 .model_id = param->params->model->model_id,
91 .el_id = param->params->model->element_idx,
92 .p_model = param->params->model
93 },
94 .evt = MESHX_BIT(event)
95 };
96
97 /* Copy the msg data from BLE Layer to MeshX Layer */
98 memcpy(&pub_param.status, &param->status_cb, sizeof(meshx_gen_client_status_cb_t));
99
100 meshx_err_t err = control_task_msg_publish(
101 CONTROL_TASK_MSG_CODE_FRM_BLE,
102 pub_param.model.model_id,
103 &pub_param,
104 sizeof(meshx_gen_cli_cb_param_t));
105 if (err)
106 {
107 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Failed to publish to control task");
108 }
109}

124meshx_err_t meshx_plat_gen_cli_init(void)
125{
126
127 meshx_err_t err = MESHX_SUCCESS;
128 if (meshx_client_init == MESHX_CLIENT_INIT_MAGIC_NO)
129 {
130 return MESHX_SUCCESS; // Already initialized
131 }
132 meshx_client_init = MESHX_CLIENT_INIT_MAGIC_NO;
133
134 // Register the callback for handling messages sent to the BLE layer
135 esp_err_t esp_err = esp_ble_mesh_register_generic_client_callback(
136 (MESHX_GEN_CLI_CB)&esp_ble_mesh_generic_client_cb);
137 if (esp_err != ESP_OK)
138 err = MESHX_ERR_PLAT;
139
140 return err;
141}

213meshx_err_t meshx_plat_gen_cli_send_msg(
214 meshx_ptr_t p_model, meshx_gen_cli_set_t *p_set,
215 uint16_t opcode, uint16_t addr,
216 uint16_t net_idx, uint16_t app_idx,
217 bool is_get_opcode
218)
219{
220 if (!p_model || !p_set)
221 {
222 return MESHX_INVALID_ARG; // Invalid arguments
223 }
224
225 esp_ble_mesh_client_common_param_t common = {0};
226 common.model = p_model;
227 common.opcode = opcode;
228 common.ctx.addr = addr;
229 common.ctx.net_idx = net_idx;
230 common.ctx.app_idx = app_idx;
231 common.ctx.send_ttl = BLE_MESH_TTL_DEFAULT;
232 common.msg_timeout = 0; /* 0 indicates that timeout value from menuconfig will be used */
233
234 esp_err_t err = ESP_OK;
235 if(!is_get_opcode)
236 {
237 err = esp_ble_mesh_generic_client_set_state(
238 &common,
239 (esp_ble_mesh_generic_client_set_state_t *)p_set
240 );
241 if (err != ESP_OK)
242 {
243 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Send Generic OnOff failed: %d", err);
244 return MESHX_FAIL;
245 }
246 }
247 else
248 {
249 err = esp_ble_mesh_generic_client_get_state(
250 &common,
251 (esp_ble_mesh_generic_client_get_state_t *)p_set
252 );
253 if (err != ESP_OK)
254 {
255 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Send Generic OnOff Get failed: %d", err);
256 return MESHX_FAIL;
257 }
258 }
259 if (err)
260 {
261 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Send Generic OnOff failed: %d", err);
262 return MESHX_FAIL;
263 }
264
265 return MESHX_SUCCESS;
266}
