67static meshx_err_t relay_server_config_srv_cb(
68 const dev_struct_t *pdev,
69 control_task_msg_evt_t evt,
70 const meshx_config_srv_cb_param_t *params)
71{
72 MESHX_UNUSED(pdev);
73 meshx_relay_srv_model_ctx_t *el_ctx = NULL;
74 size_t rel_el_id = 0;
75 uint16_t element_id = 0;
76 uint16_t base_el_id = 0;
77 bool nvs_save = false;
78 meshx_get_base_element_id(&base_el_id);
79 MESHX_LOGD(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "EVT: %d", evt);
80 switch (evt)
81 {
82 case CONTROL_TASK_MSG_EVT_APP_KEY_BIND:
83 element_id = params->state_change.mod_app_bind.element_addr - base_el_id;
84 MESHX_LOGD(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "ele: %d", element_id);
85 if (!IS_EL_IN_RANGE(element_id))
86 break;
87 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
88 el_ctx = relay_element_init_ctrl.el_list[rel_el_id].srv_ctx;
89 el_ctx->app_id = params->state_change.appkey_add.app_idx;
90 nvs_save = true;
91 break;
92 case CONTROL_TASK_MSG_EVT_PUB_ADD:
93 case CONTROL_TASK_MSG_EVT_PUB_DEL:
94 element_id = params->state_change.mod_pub_set.element_addr - base_el_id;
95 MESHX_LOGD(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "ele: %d", element_id);
96 if (!IS_EL_IN_RANGE(element_id))
97 break;
98 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
99 el_ctx = relay_element_init_ctrl.el_list[rel_el_id].srv_ctx;
100 el_ctx->pub_addr = evt == CONTROL_TASK_MSG_EVT_PUB_ADD ? params->state_change.mod_pub_set.pub_addr
101 : MESHX_ADDR_UNASSIGNED;
102 el_ctx->app_id = params->state_change.mod_pub_set.app_idx;
103 MESHX_LOGI(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "PUB_ADD: %d, %d, 0x%x, 0x%x", element_id, rel_el_id, el_ctx->pub_addr, el_ctx->app_id);
104 nvs_save = true;
105 break;
106 default:
107 break;
108 }
109 if (nvs_save)
110 {
111 meshx_err_t err = meshx_nvs_element_ctx_set(element_id, el_ctx, sizeof(meshx_relay_srv_model_ctx_t));
112 if (err != MESHX_SUCCESS)
113 {
114 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to set relay server element context: (%d)", err);
115 }
116 }
117 return MESHX_SUCCESS;
118}

135static meshx_err_t meshx_element_struct_init(uint16_t n_max)
136{
137 if (!n_max)
138 return MESHX_INVALID_ARG;
139
140 if (relay_element_init_ctrl.el_list)
141 {
142 MESHX_LOGW(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Relay element list already initialized");
143 return MESHX_INVALID_STATE;
144 }
145
146 meshx_err_t err = MESHX_SUCCESS;
147
148 relay_element_init_ctrl.element_cnt = n_max;
149 relay_element_init_ctrl.element_id_end = 0;
150 relay_element_init_ctrl.element_id_start = 0;
151
152 relay_element_init_ctrl.el_list =
153 (meshx_relay_element_t *)MESHX_CALOC(relay_element_init_ctrl.element_cnt, sizeof(meshx_relay_element_t));
154
155 if (!relay_element_init_ctrl.el_list)
156 return MESHX_NO_MEM;
157
158 for (size_t i = 0; i < relay_element_init_ctrl.element_cnt; i++)
159 {
160 relay_element_init_ctrl.el_list[i].srv_ctx =
161 (meshx_relay_srv_model_ctx_t *)MESHX_CALOC(1, sizeof(meshx_relay_srv_model_ctx_t));
162
163 if (!relay_element_init_ctrl.el_list[i].srv_ctx)
164 return MESHX_NO_MEM;
165
166 err = meshx_on_off_server_create(&relay_element_init_ctrl.el_list[i].onoff_srv_model,
167 &relay_element_init_ctrl.el_list[i].relay_srv_model_list[RELAY_SIG_ONOFF_MODEL_ID]);
168 if (err)
169 {
170 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Meshx On Off Server create failed (Err : 0x%x)", err);
171 return err;
172 }
173 relay_element_init_ctrl.el_list[i].onoff_srv_model->meshx_server_sig_model
174 = &relay_element_init_ctrl.el_list[i].relay_srv_model_list[RELAY_SIG_ONOFF_MODEL_ID];
175 }
176
177 return err;
178}

254static meshx_err_t meshx_restore_model_states(uint16_t element_id)
255{

258 meshx_relay_srv_model_ctx_t const *el_ctx = relay_element_init_ctrl.el_list[element_id].srv_ctx;

292static meshx_err_t meshx_add_relay_srv_model_to_element_list(dev_struct_t *pdev, uint16_t *start_idx, uint16_t n_max)
293{
294 if (!pdev)
295 return MESHX_INVALID_STATE;
296
297 if ((n_max + *start_idx) > CONFIG_MAX_ELEMENT_COUNT)
298 {
299 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "No of elements limit reached");
300 return MESHX_NO_MEM;
301 }
302
303 meshx_err_t err = MESHX_SUCCESS;
304 relay_element_init_ctrl.element_id_start = *start_idx;
305
306 for (uint16_t i = *start_idx; i < (n_max + *start_idx); i++)
307 {
308
309 if (i == 0)
310 continue;
311 err = meshx_plat_add_element_to_composition(
312 i,
313 pdev->elements,
314 relay_element_init_ctrl.el_list[i - *start_idx].relay_srv_model_list,
315 NULL,
316 RELAY_SRV_MODEL_SIG_CNT,
317 RELAY_SRV_MODEL_VEN_CNT);
318 if (err)
319 {
320 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to add element to composition: (%d)", err);
321 return err;
322 }
323 err = meshx_nvs_element_ctx_get(
324 i,
325 relay_element_init_ctrl.el_list[i - *start_idx].srv_ctx,
326 sizeof(meshx_relay_srv_model_ctx_t));
327 if (err != MESHX_SUCCESS)
328 {
329 MESHX_LOGW(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to get relay element context: (0x%x)", err);
330 }
331 else
332 {
333 err = meshx_restore_model_states(i - *start_idx);
334 if (err != MESHX_SUCCESS)
335 {
336 MESHX_LOGW(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to restore relay model states: (0x%x)", err);
337 }
338 }
339 }
340 relay_element_init_ctrl.element_id_end = (*start_idx += n_max);
341 return MESHX_SUCCESS;
342}

354static meshx_err_t meshx_api_control_task_handler(dev_struct_t const *pdev, control_task_msg_evt_t evt, void *params)
355{
356 MESHX_UNUSED(pdev);
357 MESHX_UNUSED(evt);
358 size_t rel_el_id = 0;
359 meshx_err_t err = MESHX_SUCCESS;
360 meshx_relay_srv_model_ctx_t *el_ctx = NULL;
361 const meshx_on_off_srv_el_msg_t *p_onoff_srv = (meshx_on_off_srv_el_msg_t *)params;
362 meshx_api_relay_server_evt_t state;
363 uint16_t element_id = p_onoff_srv->model.el_id;
364
365 if (!IS_EL_IN_RANGE(element_id))
366 return MESHX_SUCCESS;
367
368 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
369 el_ctx = relay_element_init_ctrl.el_list[rel_el_id].srv_ctx;
370
371 el_ctx->state.on_off = p_onoff_srv->on_off_state;
372
373 err = meshx_nvs_element_ctx_set(element_id, el_ctx, sizeof(meshx_relay_srv_model_ctx_t));
374 if (err != MESHX_SUCCESS)
375 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to set relay element context: (%d)", err);
376
377 state.on_off = el_ctx->state.on_off;
378 err = meshx_send_msg_to_app(
379 element_id,
380 MESHX_ELEMENT_TYPE_RELAY_SERVER,
381 RELAY_SIG_ONOFF_MODEL_ID,
382 sizeof(meshx_api_relay_server_evt_t),
383 &state);
384 if (err != MESHX_SUCCESS)
385 MESHX_LOGE(MODULE_ID_ELEMENT_SWITCH_RELAY_SERVER, "Failed to send relay state change message: (%d)", err);
386
387 return MESHX_SUCCESS;
388}

400{
401 MESHX_UNUSED(evt);

403
404 size_t rel_el_id = 0;
405 meshx_gen_srv_cb_param_t gen_srv_send;

407

416 gen_srv_send.ctx.p_ctx = NULL;
417
418 gen_srv_send.model.el_id = (uint16_t)el_id;

423 if (gen_srv_send.ctx.dst_addr == ESP_BLE_MESH_ADDR_UNASSIGNED || gen_srv_send.ctx.app_idx == ESP_BLE_MESH_KEY_UNUSED)

427 CONTROL_TASK_MSG_EVT_TO_BLE_SET_ON_OFF_SRV,
428 &gen_srv_send

431 {
