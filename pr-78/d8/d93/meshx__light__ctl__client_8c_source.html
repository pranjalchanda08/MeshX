39static meshx_err_t meshx_ctl_state_change_notify(meshx_gen_light_cli_cb_param_t *param)
40{
41 if (!param)
42 return MESHX_INVALID_ARG;
43
44 meshx_ctl_cli_el_msg_t el_light_ctl_param = {
45 .err_code = MESHX_SUCCESS,
46 .ctx = param->ctx,
47 .model = param->model
48 };
49 switch (param->ctx.opcode)
50 {
51 case MESHX_MODEL_OP_LIGHT_CTL_STATUS:
52 el_light_ctl_param.ctl_state.lightness = param->status.ctl_status.present_ctl_lightness;
53 el_light_ctl_param.ctl_state.temperature = param->status.ctl_status.present_ctl_temperature;
54 break;
55 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS:
56 el_light_ctl_param.ctl_state.delta_uv = param->status.ctl_temperature_status.present_ctl_delta_uv;
57 el_light_ctl_param.ctl_state.temperature = param->status.ctl_temperature_status.present_ctl_temperature;
58 break;
59 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS:
60 el_light_ctl_param.ctl_state.delta_uv = param->status.ctl_temperature_range_status.range_max;
61 el_light_ctl_param.ctl_state.temperature = param->status.ctl_temperature_range_status.range_min;
62 break;
63 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS:
64 el_light_ctl_param.ctl_state.delta_uv = param->status.ctl_default_status.delta_uv;
65 el_light_ctl_param.ctl_state.lightness = param->status.ctl_default_status.lightness;
66 el_light_ctl_param.ctl_state.temperature = param->status.ctl_default_status.temperature;
67 break;
68
69 default:
70 break;
71 }
72 if(param->evt == MESHX_GEN_LIGHT_CLI_TIMEOUT)
73 {
74 el_light_ctl_param.err_code = MESHX_TIMEOUT;
75 }
76
77 return control_task_msg_publish(
78 CONTROL_TASK_MSG_CODE_EL_STATE_CH,
79 CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_CTL,
80 &el_light_ctl_param,
81 sizeof(meshx_ctl_cli_el_msg_t));
82
83}

147static meshx_err_t meshx_handle_gen_light_msg(
148 const dev_struct_t *pdev,
149 control_task_msg_evt_t model_id,
150 meshx_gen_light_cli_cb_param_t *param
151)
152{
153 if (model_id != MESHX_MODEL_ID_LIGHT_CTL_CLI || !pdev || !param)
154 return MESHX_INVALID_ARG;
155
156 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "op|src|dst:%04" PRIx32 "|%04x|%04x",
157 param->ctx.opcode, param->ctx.src_addr, param->ctx.dst_addr);
158 meshx_err_t err = MESHX_SUCCESS;
159
160 switch (param->evt)
161 {
162 case MESHX_GEN_LIGHT_CLI_EVT_GET:
163 case MESHX_GEN_LIGHT_CLI_EVT_SET:
164 case MESHX_GEN_LIGHT_CLI_PUBLISH:
165 err = meshx_ctl_state_change_notify(param);
166 if (err)
167 {
168 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Failed to notify state change: %d", err);
169 }
170 break;
171 case MESHX_GEN_LIGHT_CLI_TIMEOUT:
172 err = meshx_light_ctl_client_timeout_handler();
173 if(err != MESHX_SUCCESS)
174 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Resend failed: %d", err);
175
176 else
177 err = meshx_ctl_state_change_notify(param);
178 break;
179 default:
180 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Unknown event: %d", param->evt);
181 err = MESHX_NOT_SUPPORTED;
182 break;
183 }
184
185 return err;
186}

334{
335 meshx_err_t err;

343 light_ctl_client_last_msg_ctx.opcode = opcode;

346 light_ctl_client_last_msg_ctx.p_model = model;
347
348 if (opcode == MESHX_MODEL_OP_LIGHT_CTL_GET)
349 {
350 err = meshx_gen_light_send_msg(
351 model->meshx_light_ctl_client_sig_model,
352 &set, opcode,
353 addr, net_idx,
354 app_idx
355 );
356 }
357 else if (opcode == MESHX_MODEL_OP_LIGHT_CTL_SET ||
358 opcode == MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK)
359 {
360 set.ctl_set.tid = tid;
361 set.ctl_set.op_en = false;
362 set.ctl_set.ctl_delta_uv = delta_uv;
363 set.ctl_set.ctl_lightness = lightness;
364 set.ctl_set.ctl_temperature = temperature;
365
366 memcpy(&light_ctl_client_last_msg_ctx.state, &set, sizeof(meshx_light_client_set_state_t));
367
368 err = meshx_gen_light_send_msg(
369 model->meshx_light_ctl_client_sig_model,
370 &set, opcode,
371 addr, net_idx,
372 app_idx
373 );
374 }
375 else{
376 err = MESHX_INVALID_ARG; // Invalid opcode
377 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid opcode for Light CTL Client: %04x", opcode);
378 }
