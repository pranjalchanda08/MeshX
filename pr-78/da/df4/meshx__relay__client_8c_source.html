51static meshx_err_t meshx_element_struct_init(uint16_t n_max)
52{
53 if (!n_max)
54 return MESHX_INVALID_ARG;
55
56 if (relay_element_init_ctrl.el_list)
57 {
58 MESHX_LOGW(MOD_SRC, "Relay element list already initialized");
59 return MESHX_INVALID_STATE;
60 }
61
62 meshx_err_t err = MESHX_SUCCESS;
63
64 relay_element_init_ctrl.element_cnt = n_max;
65 relay_element_init_ctrl.element_id_end = 0;
66 relay_element_init_ctrl.element_id_start = 0;
67
68 relay_element_init_ctrl.el_list =
69 (relay_client_elements_t *)MESHX_CALOC(relay_element_init_ctrl.element_cnt, sizeof(relay_client_elements_t));
70
71 if (!relay_element_init_ctrl.el_list)
72 return MESHX_NO_MEM;
73
74 for (size_t i = 0; i < relay_element_init_ctrl.element_cnt; i++)
75 {
76 RELAY_CLI_EL(i).cli_ctx =
77 (meshx_relay_client_model_ctx_t *)MESHX_CALOC(1, sizeof(meshx_relay_client_model_ctx_t));
78
79 if (!RELAY_CLI_EL(i).cli_ctx)
80 return MESHX_NO_MEM;
81
82 err = meshx_on_off_client_create(&RELAY_CLI_EL(i).onoff_cli_model,
83 &RELAY_CLI_EL(i).relay_cli_sig_model_list[RELAY_CLI_SIG_ONOFF_MODEL_ID]);
84 if (err)
85 {
86 MESHX_LOGE(MOD_SRC, "Meshx On Off Client create failed (Err : 0x%x)", err);
87 return err;
88 }
89 RELAY_CLI_EL(i).onoff_cli_model->meshx_onoff_client_sig_model
90 = &RELAY_CLI_EL(i).relay_cli_sig_model_list[RELAY_CLI_SIG_ONOFF_MODEL_ID];
91 }
92
93 return MESHX_SUCCESS;
94}

169static meshx_err_t meshx_add_relay_cli_model_to_element_list(
170 dev_struct_t *pdev, uint16_t *start_idx, uint16_t n_max)
171{
172 if (!pdev || !start_idx || !n_max)
173 return MESHX_INVALID_STATE;
174
175 if ((n_max + *start_idx) > CONFIG_MAX_ELEMENT_COUNT)
176 {
177 MESHX_LOGE(MOD_SRC, "No of elements limit reached");
178 return MESHX_NO_MEM;
179 }
180
181 meshx_err_t err = MESHX_SUCCESS;
182 relay_element_init_ctrl.element_id_start = *start_idx;
183
184 for (uint16_t i = *start_idx; i < (n_max + *start_idx); i++)
185 {
186 if (i == 0)
187 continue;
188 err = meshx_plat_add_element_to_composition(
189 i,
190 pdev->elements,
191 RELAY_CLI_EL(i - *start_idx).relay_cli_sig_model_list,
192 NULL,
193 RELAY_CLI_MODEL_SIG_CNT,
194 RELAY_CLI_MODEL_VEN_CNT);
195 if (err)
196 {
197 MESHX_LOGE(MOD_SRC, "Failed to add element to composition: (%d)", err);
198 return err;
199 }
200 err = meshx_nvs_element_ctx_get(
201 i,
202 RELAY_CLI_EL(i - *start_idx).cli_ctx,
203 sizeof(meshx_relay_client_model_ctx_t));
204 if (err != MESHX_SUCCESS)
205 {
206 MESHX_LOGW(MOD_SRC, "Failed to get relay cli element context: (0x%x)", err);
207 }
208 }
209 /* Increment the index for further registrations */
210 relay_element_init_ctrl.element_id_end = *start_idx += n_max;
211 return MESHX_SUCCESS;
212}

226static meshx_err_t relay_client_config_cli_cb(
227 const dev_struct_t *pdev,
228 control_task_msg_evt_t evt,
229 const meshx_config_srv_cb_param_t *params)
230{
231 MESHX_UNUSED(pdev);
232 meshx_relay_client_model_ctx_t *el_ctx = NULL;
233 size_t rel_el_id = 0;
234 uint16_t element_id = 0;
235 uint16_t base_el_id = 0;
236 bool nvs_save = false;
237 meshx_get_base_element_id(&base_el_id);
238
239 MESHX_LOGD(MODULE_ID_ELEMENT_SWITCH_RELAY_CLIENT, "EVT: %p", (void *)evt);
240 switch (evt)
241 {
242 case CONTROL_TASK_MSG_EVT_APP_KEY_BIND:
243 element_id = params->state_change.mod_app_bind.element_addr - base_el_id;
244 if (!IS_EL_IN_RANGE(element_id))
245 break;
246 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
247 el_ctx = RELAY_CLI_EL(rel_el_id).cli_ctx;
248 el_ctx->app_id = params->state_change.mod_app_bind.app_idx;
249 nvs_save = true;
250 break;
251 case CONTROL_TASK_MSG_EVT_PUB_ADD:
252 case CONTROL_TASK_MSG_EVT_PUB_DEL:
253 element_id = params->state_change.mod_pub_set.element_addr - base_el_id;
254 if (!IS_EL_IN_RANGE(element_id))
255 break;
256 rel_el_id = GET_RELATIVE_EL_IDX(element_id);
257 el_ctx = RELAY_CLI_EL(rel_el_id).cli_ctx;
258 el_ctx->pub_addr = evt == CONTROL_TASK_MSG_EVT_PUB_ADD ? params->state_change.mod_pub_set.pub_addr
259 : MESHX_ADDR_UNASSIGNED;
260 el_ctx->app_id = params->state_change.mod_pub_set.app_idx;
261 nvs_save = true;
262 MESHX_LOGI(MOD_SRC, "PUB_ADD: %d, %d, 0x%x, 0x%x", element_id, rel_el_id, el_ctx->pub_addr, el_ctx->app_id);
263 break;
264 default:
265 break;
266 }
267 if (nvs_save)
268 {
269 meshx_err_t err = meshx_nvs_element_ctx_set(element_id, el_ctx, sizeof(meshx_relay_client_model_ctx_t));
270 if (err != MESHX_SUCCESS)
271 {
272 MESHX_LOGE(MOD_SRC, "Failed to set relay element context: (%d)", err);
273 }
274 }
275 return MESHX_SUCCESS;
276}

326static meshx_err_t meshx_relay_cli_send_onoff_msg(
327 const dev_struct_t *pdev,
328 uint16_t element_id,
329 uint8_t set_get,
330 uint8_t ack)
331{
332 if (!pdev || !IS_EL_IN_RANGE(element_id))
333 return MESHX_INVALID_ARG;
334
335 meshx_err_t err = MESHX_SUCCESS;
336 size_t rel_el_id = GET_RELATIVE_EL_IDX(element_id);
337 meshx_onoff_client_model_t *model = RELAY_CLI_EL(rel_el_id).onoff_cli_model;
338
339 meshx_relay_client_model_ctx_t *el_ctx = RELAY_CLI_EL(rel_el_id).cli_ctx;
340 uint16_t opcode = MESHX_MODEL_OP_GEN_ONOFF_GET;
341
342 if (MESHX_GEN_ON_OFF_CLI_MSG_SET == set_get)
343 {
344 opcode = ack ? MESHX_MODEL_OP_GEN_ONOFF_SET : MESHX_MODEL_OP_GEN_ONOFF_SET_UNACK;
345 }
346
347 MESHX_LOGD(MOD_SRC, "OPCODE: %p", (void *)(uint32_t)opcode);
348
349 /* Send message to the relay client */
350 err = meshx_onoff_client_send_msg(
351 model,
352 opcode,
353 el_ctx->pub_addr,
354 pdev->meshx_store.net_key_id,
355 el_ctx->app_id,
356 el_ctx->state.on_off,
357 el_ctx->tid
358 );
359 if (err)
360 {
361 MESHX_LOGE(MOD_SRC, "Relay Client Send Message failed: (%d)", err);
362 }
363 else
364 {
365 el_ctx->tid++;
366 if(opcode == MESHX_MODEL_OP_GEN_ONOFF_SET_UNACK)
367 {
368 el_ctx->state.prev_on_off = el_ctx->state.on_off;
369 el_ctx->state.on_off = !el_ctx->state.on_off;
370 }
371 }
372 return err;
373}

387static meshx_err_t meshx_relay_client_element_state_change_handler(
388 const dev_struct_t *pdev,
389 control_task_msg_evt_t evt,
390 const meshx_on_off_cli_el_msg_t *param
391)
392{
393 MESHX_UNUSED(pdev);
394 MESHX_UNUSED(evt);
395 uint16_t element_id = param->model.el_id;
396 if (!IS_EL_IN_RANGE(element_id))
397 {
398 return MESHX_SUCCESS;
399 }
400
401 size_t rel_el_id = GET_RELATIVE_EL_IDX(element_id);
402 meshx_relay_client_model_ctx_t *el_ctx = RELAY_CLI_EL(rel_el_id).cli_ctx;
403 meshx_api_relay_client_evt_t app_notify;
404 meshx_err_t err = MESHX_SUCCESS;
405 bool nvs_save = false;
406 if(param->err_code == MESHX_SUCCESS)
407 {
408
409 RELAY_CLI_EL(rel_el_id).element_model_init |= MESHX_BIT(RELAY_CLI_SIG_ONOFF_MODEL_ID);
410 if (el_ctx->state.prev_on_off != param->on_off_state)
411 {
412 el_ctx->state.prev_on_off = param->on_off_state;
413 app_notify.err_code = 0;
414 app_notify.on_off = el_ctx->state.prev_on_off;
415
416 err = meshx_send_msg_to_app(element_id,
417 MESHX_ELEMENT_TYPE_RELAY_CLIENT,
418 MESHX_ELEMENT_FUNC_ID_RELAY_SERVER_ONN_OFF,
419 sizeof(meshx_api_relay_client_evt_t),
420 &app_notify);
421 if (err != MESHX_SUCCESS)
422 {
423 MESHX_LOGE(MOD_SRC, "Failed to send relay state change message: (%d)", err);
424 }
425 }
426
427 else
428 {
429 MESHX_LOGD(MOD_SRC, "No change in state: %d", param->on_off_state);
430 }
431
432 el_ctx->state.on_off = !param->on_off_state;
433 MESHX_LOGD(MOD_SRC, "SET|PUBLISH: %d", param->on_off_state);
434 MESHX_LOGD(MOD_SRC, "Next state: %d", el_ctx->state.on_off);
435 nvs_save = true;
436 }
437 else
438 {
439 MESHX_LOGE(MOD_SRC, "Relay Client Element Message: Error (%d)", param->err_code);
440 }
441 if (nvs_save)
442 {
443 err = meshx_nvs_element_ctx_set(element_id, el_ctx, sizeof(meshx_relay_client_model_ctx_t));
444 if (err != MESHX_SUCCESS)
445 {
446 MESHX_LOGE(MOD_SRC, "Failed to set relay element context: (%d)", err);
447 }
448 }
449 return MESHX_SUCCESS;
450}

463static meshx_err_t meshx_relay_client_to_ble_handler(

487 MESHX_LOGE(

489 "Relay Client Control Task: Set OnOff failed (%p)", (void *)err);
490 }
491 }
492 return err;
