34static meshx_err_t ut_command_handler(int argc, char **argv) {
35 if (argc < UT_CMD_MIN_ARGS) {
36 ESP_LOGE(TAG, "Insufficient arguments");
37 return MESHX_INVALID_ARG;
38 }
39
40 int cmd_id = UT_GET_ARG(2, uint16_t, argv);
41 int parsed_argc = UT_GET_ARG(3, uint16_t, argv);
42 module_id_t module_id = UT_GET_ARG(1, uint16_t, argv);
43 ESP_LOGD(TAG, "Unit Test: Params -> argc: %d, Module: %d, cmd_id: %d", parsed_argc, cmd_id , module_id);
44 if (parsed_argc > (argc - UT_CMD_MIN_ARGS))
45 {
46 ESP_LOGE(TAG, "Insufficient module arguments");
47 return MESHX_INVALID_ARG;
48 }
49
50 for (size_t i = 0; i < parsed_argc; i++)
51 {
52 ESP_LOGD(TAG, "argv[%d]: %s", i, argv[i + UT_CMD_MIN_ARGS]);
53 }
54
55 if (module_id >= MODULE_ID_MAX) {
56 ESP_LOGE(TAG, "Module ID %d unknown", module_id);
57 return MESHX_INVALID_ARG;
58 }
59
60 if(NULL != callback_list[module_id].callback)
61 {
62 return callback_list[module_id].callback(cmd_id, parsed_argc, (argv + UT_CMD_MIN_ARGS));
63 }
64
65 ESP_LOGE(TAG, "No unit test registered for module ID %d", module_id);
66 return MESHX_NOT_FOUND;
67}
