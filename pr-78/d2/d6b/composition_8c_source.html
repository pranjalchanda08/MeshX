132static meshx_err_t meshx_prov_control_task_handler(dev_struct_t *pdev, control_task_msg_evt_t evt, void *params)
133{
134 const meshx_prov_cb_param_t *param = (meshx_prov_cb_param_t*) params;
135
136 switch (evt)
137 {
138 case CONTROL_TASK_MSG_EVT_PROVISION_STOP:
139 pdev->meshx_store.net_key_id = param->node_prov_complete.net_idx;
140 pdev->meshx_store.node_addr = param->node_prov_complete.addr;
141 meshx_nvs_set(MESHX_NVS_STORE, &pdev->meshx_store, sizeof(pdev->meshx_store), MESHX_NVS_AUTO_COMMIT);
142 break;
143 case CONTROL_TASK_MSG_EVT_IDENTIFY_START:
144 MESHX_LOGI(MODULE_ID_COMMON, "Identify Start");
145 break;
146 default:
147 break;
148 }
149 return MESHX_SUCCESS;
150}

157MESHX_MODEL * get_root_sig_models(void)
158{
159 static MESHX_MODEL temp_model;
160 if(meshx_sig_root_model_arr == NULL && meshx_sig_root_model_arr_len)
161 {
162 meshx_sig_root_model_arr = (MESHX_MODEL *) MESHX_MALLOC(sizeof(MESHX_MODEL) * meshx_sig_root_model_arr_len);
163 if(meshx_sig_root_model_arr == NULL)

166 return NULL;

168 memset(meshx_sig_root_model_arr, 0, sizeof(MESHX_MODEL) * meshx_sig_root_model_arr_len);
169
170 for(uint16_t i = 0; i < meshx_sig_root_model_arr_len; i++)
171 {
172 if(meshx_sig_root_model_getfn[i] == NULL)
173 {
174 continue;
175 }
176 meshx_sig_root_model_getfn[i]((void**)&temp_model);
177 memcpy(&meshx_sig_root_model_arr[i], &temp_model, sizeof(MESHX_MODEL));
178 }
179 }

244meshx_err_t meshx_create_element_composition(dev_struct_t *p_dev, meshx_config_t const *config)
245{
246#if CONFIG_MAX_ELEMENT_COUNT > 0
247 meshx_err_t err;
248#if CONFIG_SECTION_ENABLE_ELEMENT_TABLE
249 element_comp_table_t *element_comp_table;
250#endif /* CONFIG_SECTION_ENABLE_ELEMENT_TABLE */
251 if(!p_dev || !config || !config->element_comp_arr_len || !config->element_comp_arr)
252 return MESHX_INVALID_ARG;
253
254 err = control_task_msg_subscribe(
255 CONTROL_TASK_MSG_CODE_PROVISION,
256 CONTROL_TASK_PROV_EVT_MASK,
257 &meshx_prov_control_task_handler);
258 MESHX_ERR_PRINT_RET("Failed to register control task callback", err);
259
260 err = meshx_init_config_server();
261 MESHX_ERR_PRINT_RET("Failed to initialize config server", err);
262 for(uint16_t element_id = 0; element_id < config->element_comp_arr_len; element_id++)
263 {
264#if !CONFIG_SECTION_ENABLE_ELEMENT_TABLE
265 if(config->element_comp_arr[element_id].element_cnt != 0
266 && element_comp_fn[config->element_comp_arr[element_id].type] != NULL)
267 {
268 err = element_comp_fn[config->element_comp_arr[element_id].type](p_dev, config->element_comp_arr[element_id].element_cnt);
269 if(err)
270 {
271 MESHX_LOGE(MODULE_ID_COMMON, "Element composition failed: (%d)", err);
272 return err;
273 }
274 }
275#else
276 element_comp_table = &MESHX_ELEMENT_COMP_TABLE_START;
277 while(element_comp_table < &MESHX_ELEMENT_COMP_TABLE_STOP)
278 {
279 if(element_comp_table->idx == element_id && config->element_comp_arr[element_id].element_cnt != 0)
280 {
281 err = element_comp_table->element_comp_fn(p_dev, config->element_comp_arr[element_id].element_cnt);
282 if(err)
283 {
284 MESHX_LOGE(MODULE_ID_COMMON, "Element composition failed: (%d)", err);
285 return err;
286 }
287 }
288 element_comp_table++;
289 }
290#endif /* CONFIG_SECTION_ENABLE_ELEMENT_TABLE */
291 }
292#endif /* CONFIG_MAX_ELEMENT_COUNT */
293 return err;
294}
