31static meshx_err_t meshx_state_change_notify(meshx_gen_cli_cb_param_t *param)
32{
33 if (!param)
34 return MESHX_INVALID_ARG;
35
36 meshx_on_off_cli_el_msg_t srv_onoff_param = {
37 .err_code = MESHX_SUCCESS,
38 .ctx = param->ctx,
39 .model = param->model,
40 .on_off_state = param->status.onoff_status.present_onoff
41 };
42 if(param->evt == MESHX_GEN_CLI_TIMEOUT)
43 {
44 srv_onoff_param.err_code = MESHX_TIMEOUT;
45 }
46
47 if (MESHX_ADDR_IS_UNICAST(param->ctx.dst_addr)
48 || (MESHX_ADDR_BROADCAST(param->ctx.dst_addr))
49 || (MESHX_ADDR_IS_GROUP(param->ctx.dst_addr)
50 && (MESHX_SUCCESS == meshx_is_group_subscribed(param->model.p_model, param->ctx.dst_addr))))
51 {
52 meshx_err_t err = control_task_msg_publish(
53 CONTROL_TASK_MSG_CODE_EL_STATE_CH,
54 CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_ON_OFF,
55 &srv_onoff_param,
56 sizeof(meshx_on_off_cli_el_msg_t));
57 return err ? err : MESHX_SUCCESS;
58 }
59 return MESHX_NOT_SUPPORTED;
60}

74static meshx_err_t meshx_handle_gen_onoff_msg(
75 const dev_struct_t *pdev,
76 control_task_msg_evt_t model_id,
77 meshx_gen_cli_cb_param_t *param
78)
79{
80 if(!param || !pdev)
81 {
82 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid parameters");
83 return MESHX_INVALID_ARG;
84 }
85 if(model_id != MESHX_MODEL_ID_GEN_ONOFF_CLI)
86 {
87 return MESHX_SUCCESS;
88 }
89 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "op|src|dst:%04" PRIx32 "|%04x|%04x",
90 param->ctx.opcode, param->ctx.src_addr, param->ctx.dst_addr);
91
92 meshx_err_t err = MESHX_SUCCESS;
93
94 switch (param->evt)
95 {
96 case MESHX_GEN_CLI_EVT_SET:
97 case MESHX_GEN_CLI_PUBLISH:
98 err = meshx_state_change_notify(param);
99 break;
100 case MESHX_GEN_CLI_TIMEOUT:
101 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Timeout");
102 err = meshx_state_change_notify(param);
103 break;
104 default:
105 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Unhandled event: %d", param->evt);
106 break;
107 }
108 return err;
109}

119meshx_err_t meshx_on_off_client_init(void)
120{
121 meshx_err_t err = MESHX_SUCCESS;
122
123 if (meshx_client_init_flag == MESHX_CLIENT_INIT_MAGIC)
124 {
125 return MESHX_SUCCESS;
126 }
127 meshx_client_init_flag = MESHX_CLIENT_INIT_MAGIC;
128
129 err = meshx_gen_client_init();
130 if (err)
131 {
132 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Failed to initialize meshx client");
133 }
134
135 err = meshx_gen_client_from_ble_reg_cb(
136 MESHX_MODEL_ID_GEN_ONOFF_CLI,
137 (meshx_gen_client_cb_t)&meshx_handle_gen_onoff_msg
138 );
139 if (err)
140 {
141 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to initialize meshx_gen_srv_reg_cb (Err: %d)", err);
142 }
143
144 return err;
145}

228meshx_err_t meshx_onoff_client_send_msg(
229 meshx_onoff_client_model_t *model,
230 uint16_t opcode, uint16_t addr,
231 uint16_t net_idx, uint16_t app_idx,
232 uint8_t state, uint8_t tid
233)
234{
235 meshx_err_t err;
236 meshx_gen_cli_set_t set = {0};
237 if (!model || !model->meshx_onoff_client_sig_model)
238 {
239 return MESHX_INVALID_ARG;
240 }
241
242 if(opcode == MESHX_MODEL_OP_GEN_ONOFF_GET)
243 {
244 err = meshx_gen_cli_send_msg(
245 model->meshx_onoff_client_sig_model,
246 &set, opcode,
247 addr, net_idx,
248 app_idx
249 );
250 }
251
252 else if (opcode == MESHX_MODEL_OP_GEN_ONOFF_SET ||
253 opcode == MESHX_MODEL_OP_GEN_ONOFF_SET_UNACK)
254 {
255 set.onoff_set.tid = tid;
256 set.onoff_set.onoff = state;
257 set.onoff_set.op_en = false;
258
259 err = meshx_gen_cli_send_msg(
260 model->meshx_onoff_client_sig_model,
261 &set, opcode,
262 addr, net_idx,
263 app_idx
264 );
265 }
266 else{
267 err = MESHX_INVALID_ARG; // Invalid opcode
268 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid opcode for Generic OnOff Client: %04x", opcode);
269 }
270 return err;
271}
