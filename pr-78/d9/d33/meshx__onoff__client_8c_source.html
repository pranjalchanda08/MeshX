65static meshx_err_t meshx_handle_gen_onoff_msg(
66 const dev_struct_t *pdev,
67 control_task_msg_evt_t model_id,
68 meshx_gen_cli_cb_param_t *param
69)
70{
71 if(!param || !pdev)
72 {
73 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid parameters");
74 return MESHX_INVALID_ARG;
75 }
76 if(model_id != MESHX_MODEL_ID_GEN_ONOFF_CLI)
77 {
78 return MESHX_SUCCESS;
79 }
80 MESHX_LOGD(MODULE_ID_MODEL_CLIENT, "op|src|dst:%04" PRIx32 "|%04x|%04x",
81 param->ctx.opcode, param->ctx.src_addr, param->ctx.dst_addr);
82
83 meshx_err_t err = MESHX_SUCCESS;
84
85 switch (param->evt)
86 {
87 case MESHX_GEN_CLI_EVT_GET:
88 case MESHX_GEN_CLI_EVT_SET:
89 case MESHX_GEN_CLI_PUBLISH:
90 err = meshx_state_change_notify(param, MESHX_SUCCESS);
91 break;
92 case MESHX_GEN_CLI_TIMEOUT:
93 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Timeout");
94 err = meshx_state_change_notify(param, MESHX_TIMEOUT);
95 break;
96 default:
97 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Unhandled event: %d", param->evt);
98 break;
99 }
100 return err;
101}

220meshx_err_t meshx_onoff_client_send_msg(
221 meshx_onoff_client_model_t *model,
222 uint16_t opcode, uint16_t addr,
223 uint16_t net_idx, uint16_t app_idx,
224 uint8_t state, uint8_t tid
225)
226{
227 meshx_err_t err;
228 meshx_gen_cli_set_t set = {0};
229 if (!model || !model->meshx_onoff_client_sig_model)
230 {
231 return MESHX_INVALID_ARG;
232 }
233
234 if(opcode == MESHX_MODEL_OP_GEN_ONOFF_GET)
235 {
236 err = meshx_gen_cli_send_msg(
237 model->meshx_onoff_client_sig_model,
238 &set, opcode,
239 addr, net_idx,
240 app_idx
241 );
242 }
243
244 else if (opcode == MESHX_MODEL_OP_GEN_ONOFF_SET ||
245 opcode == MESHX_MODEL_OP_GEN_ONOFF_SET_UNACK)
246 {
247 set.onoff_set.tid = tid;
248 set.onoff_set.onoff = state;
249 set.onoff_set.op_en = false;
250
251 err = meshx_gen_cli_send_msg(
252 model->meshx_onoff_client_sig_model,
253 &set, opcode,
254 addr, net_idx,
255 app_idx
256 );
257 }
258 else{
259 err = MESHX_INVALID_ARG; // Invalid opcode
260 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid opcode for Generic OnOff Client: %04x", opcode);
261 }
262 return err;
