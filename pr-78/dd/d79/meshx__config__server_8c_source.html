51static meshx_err_t meshx_config_server_control_task_handler(
52 const dev_struct_t *pdev,
53 control_task_msg_evt_t evt,
54 const void *params
55)
56{
57 if(!pdev || !params)
58 {
59 return MESHX_INVALID_ARG;
60 }
61 const meshx_config_srv_cb_param_t *pub_param = (const meshx_config_srv_cb_param_t *)params;
62
63 /* Map event to config event */
64 config_evt_t config_evt = CONTROL_TASK_MSG_EVT_CONFIG_ALL;
65 for (uint16_t i = 0; i < config_srv_evt_map_count; i++)
66 {
67 if (config_server_model_evt_map_table[i].model_op_code == pub_param->ctx.opcode)
68 {
69 config_evt = config_server_model_evt_map_table[i].config_evt;
70 break;
71 }
72 }
73 if (config_evt == CONTROL_TASK_MSG_EVT_CONFIG_ALL)
74 {
75 return MESHX_INVALID_ARG;
76 }
77 /* Publish the event */
78 return control_task_msg_publish(
79 CONTROL_TASK_MSG_CODE_CONFIG,
80 config_evt,
81 pub_param,
82 sizeof(meshx_config_srv_cb_param_t));
83}
