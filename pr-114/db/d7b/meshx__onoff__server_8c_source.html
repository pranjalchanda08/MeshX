27static meshx_err_t meshx_state_change_notify(meshx_gen_srv_cb_param_t *param)
28{
29 if (!param)
30 return MESHX_INVALID_ARG;
31
32 meshx_on_off_srv_el_msg_t srv_onoff_param = {
33 .model = param->model,
34 .on_off_state = param->state_change.onoff_set.onoff};
35
36 if (MESHX_ADDR_IS_UNICAST(param->ctx.dst_addr) || (MESHX_ADDR_BROADCAST(param->ctx.dst_addr)) || (MESHX_ADDR_IS_GROUP(param->ctx.dst_addr) && (MESHX_SUCCESS == meshx_is_group_subscribed(param->model.p_model, param->ctx.dst_addr))))
37 {
38 meshx_err_t err = control_task_msg_publish(
39 CONTROL_TASK_MSG_CODE_EL_STATE_CH,
40 CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_ON_OFF,
41 &srv_onoff_param,
42 sizeof(meshx_on_off_srv_el_msg_t));
43 return err ? err : MESHX_SUCCESS;
44 }
45 return MESHX_NOT_SUPPORTED;
46}

64static meshx_err_t meshx_handle_gen_onoff_msg(const dev_struct_t *pdev, control_task_msg_evt_t model_id, meshx_gen_srv_cb_param_t *param)
65{
66 MESHX_UNUSED(pdev);
67 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "op|src|dst:%04" PRIx32 "|%04x|%04x",
68 param->ctx.opcode, param->ctx.src_addr, param->ctx.dst_addr);
69 if (model_id != MESHX_MODEL_ID_GEN_ONOFF_SRV)
70 return MESHX_INVALID_ARG;
71
72 bool send_reply = (param->ctx.opcode != MESHX_MODEL_OP_GEN_ONOFF_SET_UNACK);
73 switch (param->ctx.opcode)
74 {
75 case MESHX_MODEL_OP_GEN_ONOFF_GET:
76 break;
77 case MESHX_MODEL_OP_GEN_ONOFF_SET:
78 case MESHX_MODEL_OP_GEN_ONOFF_SET_UNACK:
79 meshx_state_change_notify(param);
80 break;
81 default:
82 break;
83 }
84 if (send_reply
85 /* This is meant to notify the respective publish client */
86 || param->ctx.src_addr != param->model.pub_addr)
87 {
88 /* Here the message was received from unregistered source and mention the state to the respective client */
89 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "PUB: src|pub %x|%x", param->ctx.src_addr, param->model.pub_addr);
90 param->ctx.opcode = MESHX_MODEL_OP_GEN_ONOFF_STATUS;
91 param->ctx.dst_addr = param->model.pub_addr;
92
93 return meshx_gen_srv_send_msg_to_ble(
94 CONTROL_TASK_MSG_EVT_TO_BLE_SET_ON_OFF_SRV,
95 param);
96 }
97 return MESHX_SUCCESS;
98}
