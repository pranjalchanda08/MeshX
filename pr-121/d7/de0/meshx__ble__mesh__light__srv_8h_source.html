5 * @brief Header file for the MeshX BLE Mesh Light Server module.
6 * This file defines the data structures, function prototypes, and
7 * callback parameters for managing the Light Server models in a BLE Mesh network.

288typedef struct {
289 /**
290 * 0b0 The controller is turned off.
291 * - The binding with the Light Lightness state is disabled.
292 * 0b1 The controller is turned on.
293 * - The binding with the Light Lightness state is enabled.
294 */
295 uint32_t mode : 1; /*!< The value of Light LC Mode state */
296 uint32_t occupancy_mode : 1; /*!< The value of Light LC Occupancy Mode state */
297 uint32_t light_onoff : 1; /*!< The present value of Light LC Light OnOff state */
298 uint32_t target_light_onoff : 1; /*!< The target value of Light LC Light OnOff state */
299 uint32_t occupancy : 1; /*!< The value of Light LC Occupancy state */
300 uint32_t ambient_luxlevel : 24; /*!< The value of Light LC Ambient LuxLevel state */
301
302 /**
303 * 1. Light LC Linear Output = max((Lightness Out)^2/65535, Regulator Output)
304 * 2. If the Light LC Mode state is set to 0b1, the binding is enabled and upon
305 * a change of the Light LC Linear Output state, the following operation
306 * shall be performed:
307 * Light Lightness Linear = Light LC Linear Output
308 * 3. If the Light LC Mode state is set to 0b0, the binding is disabled (i.e.,
309 * upon a change of the Light LC Linear Output state, no operation on the
310 * Light Lightness Linear state is performed).
311 */
312 uint16_t linear_output; /*!< The value of Light LC Linear Output state */
313} meshx_light_lc_state_t;
