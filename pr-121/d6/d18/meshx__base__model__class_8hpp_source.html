31/**
32 * @class meshXBaseModel

34 *
35 * This class serves as the foundation for all MeshX models, providing common
36 * functionality and interfaces for both client and server models.
37 *
38 * @tparam meshXBaseT Template parameter for base model customization.
39 * This should be a derived class that implements platform-specific behavior.
40 * @tparam ble_mesh_send_msg_params_t BLE mesh message parameters type.
41 * This type defines the structure used for sending BLE mesh messages.
42 *
43 * @note This is an abstract base class that cannot be instantiated directly.
44 * Use meshXBaseServerModel or meshXBaseClientModel instead.

85 /**
86 * @brief Send message through the model
87 * @details Pure virtual function to be implemented by derived classes for sending messages
88 * through the BLE mesh network. The implementation should handle the actual
89 * transmission of the message to the BLE stack.
90 *
91 * @tparam meshXBaseT Derived model type
92 * @tparam ble_mesh_send_msg_params_t Type for BLE mesh send message parameters
93 *
94 * @param[in] params Pointer to message parameters structure containing:
95 * - opcode: Message opcode
96 * - msg: Pointer to message data
97 * - len: Length of message data
98 * - dst: Destination address
99 * - app_idx: Application key index
100 * - ttl: Time to live value
101 *
102 * @return MESHX_SUCCESS on success, error code otherwise
103 * @retval MESHX_SUCCESS Message sent successfully
104 * @retval MESHX_ERR_INVALID_ARG Invalid parameters
105 * @retval MESHX_ERR_NO_MEM Insufficient memory
106 * @retval MESHX_ERR_INTERNAL Internal error
107 */
108 virtual meshx_err_t plat_send_msg(ble_mesh_send_msg_params_t *params) = 0;
