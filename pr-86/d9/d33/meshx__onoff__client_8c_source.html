100static meshx_err_t meshx_handle_gen_onoff_msg(
101 const dev_struct_t *pdev,
102 control_task_msg_evt_t model_id,
103 meshx_gen_cli_cb_param_t *param
104)
105{
106 if(!param || !pdev)
107 {
108 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid parameters");
109 return MESHX_INVALID_ARG;
110 }
111 if(model_id != MESHX_MODEL_ID_GEN_ONOFF_CLI)
112 {
113 return MESHX_SUCCESS;
114 }
115 MESHX_LOGD(MODULE_ID_MODEL_CLIENT, "op|src|dst:%04" PRIx32 "|%04x|%04x",
116 param->ctx.opcode, param->ctx.src_addr, param->ctx.dst_addr);
117
118 meshx_err_t err = MESHX_SUCCESS;
119
120 switch (param->evt)
121 {
122 case MESHX_GEN_CLI_EVT_GET:
123 case MESHX_GEN_CLI_EVT_SET:
124 case MESHX_GEN_CLI_PUBLISH:
125 err = meshx_state_change_notify(param, MESHX_SUCCESS);
126 break;
127 case MESHX_GEN_CLI_TIMEOUT:
128 err = meshx_relay_client_timeout_handler();
129 if(err != MESHX_SUCCESS)
130 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Resend failed: %d", err);
131 else
132 err = meshx_state_change_notify(param, MESHX_TIMEOUT);
133 break;
134 default:
135 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Unhandled event: %d", param->evt);
136 break;
137 }
138 return err;
139}

149meshx_err_t meshx_on_off_client_init(void)
150{
151 meshx_err_t err = MESHX_SUCCESS;
152
153 if (meshx_client_init_flag == MESHX_CLIENT_INIT_MAGIC)
154 {
155 return MESHX_SUCCESS;
156 }
157 meshx_client_init_flag = MESHX_CLIENT_INIT_MAGIC;
158
159 err = meshx_gen_client_init();
160 if (err)
161 {
162 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Failed to initialize meshx client");
163 }
164
165 err = meshx_gen_client_from_ble_reg_cb(
166 MESHX_MODEL_ID_GEN_ONOFF_CLI,
167 (meshx_gen_client_cb_t)&meshx_handle_gen_onoff_msg
168 );
169 if (err)
170 {
171 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to initialize meshx_gen_srv_reg_cb (Err: %d)", err);
172 }
173
174 return err;
175}

258meshx_err_t meshx_onoff_client_send_msg(
259 meshx_onoff_client_model_t *model,
260 uint16_t opcode, uint16_t addr,
261 uint16_t net_idx, uint16_t app_idx,
262 uint8_t state, uint8_t tid
263)
264{
265 meshx_err_t err;
266 meshx_gen_cli_set_t set = {0};
267 if (!model || !model->meshx_onoff_client_sig_model)
268 {
269 return MESHX_INVALID_ARG;
270 }
271
272 onoff_client_last_msg_ctx.addr = addr;
273 onoff_client_last_msg_ctx.opcode = opcode;
274 onoff_client_last_msg_ctx.p_model = model;
275 onoff_client_last_msg_ctx.net_idx = net_idx;
276 onoff_client_last_msg_ctx.app_idx = app_idx;
277
278 if(opcode == MESHX_MODEL_OP_GEN_ONOFF_GET)
279 {
280 err = meshx_gen_cli_send_msg(
281 model->meshx_onoff_client_sig_model,
282 &set, opcode,
283 addr, net_idx,
284 app_idx
285 );
286 }
287
288 else if (opcode == MESHX_MODEL_OP_GEN_ONOFF_SET ||
289 opcode == MESHX_MODEL_OP_GEN_ONOFF_SET_UNACK)
290 {
291 set.onoff_set.tid = tid;
292 set.onoff_set.onoff = state;
293 set.onoff_set.op_en = false;
294
295 memcpy(&onoff_client_last_msg_ctx.state, &set, sizeof(meshx_gen_cli_set_t));
296
297 err = meshx_gen_cli_send_msg(
298 model->meshx_onoff_client_sig_model,
299 &set, opcode,
300 addr, net_idx,
301 app_idx
302 );
303 }
304 else{
305 err = MESHX_INVALID_ARG; // Invalid opcode
306 MESHX_LOGE(MODULE_ID_MODEL_CLIENT, "Invalid opcode for Generic OnOff Client: %04x", opcode);
307 }
308 return err;
309}
