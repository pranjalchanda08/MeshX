75static void meshx_provisioning_cb(esp_ble_mesh_prov_cb_event_t event,
76 const esp_ble_mesh_prov_cb_param_t *param)
77{
78 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "Event 0x%02x", event);
79
80 if (prov_cb_evt_ctrl_task_evt_table[event].ctrl_task_evt != 0)
81 {
82 /* Entry available in prov_cb_evt_ctrl_task_evt_table */
83 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "%s", prov_cb_evt_ctrl_task_evt_table[event].evt_str);
84 if (send_control_msg(param, prov_cb_evt_ctrl_task_evt_table[event].ctrl_task_evt) != MESHX_SUCCESS)
85 {
86 MESHX_LOGE(MODULE_ID_MODEL_SERVER, "Failed to send control message");
87 }
88 }
89 else
90 {
91 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "Unhandled event: %d", event);
92 }
93 if (event == ESP_BLE_MESH_NODE_PROV_COMPLETE_EVT)
94 {
95 MESHX_LOGI(MODULE_ID_MODEL_SERVER, "net_idx: 0x%04x, addr: " LOG_ANSI_COLOR_REGULAR(LOG_ANSI_COLOR_CYAN) "0x%04x" LOG_ANSI_COLOR_RESET, param->node_prov_complete.net_idx, param->node_prov_complete.addr);
96 MESHX_LOGI(MODULE_ID_MODEL_SERVER, "flags: 0x%02x, iv_index: 0x%08" PRIx32, param->node_prov_complete.flags, param->node_prov_complete.iv_index);
97 }
98}
