79static meshx_err_t meshx_element_init(dev_struct_t *p_dev, meshx_config_t const *config)
80{
81 if (!p_dev)
82 return MESHX_INVALID_STATE;
83
84 meshx_err_t err = MESHX_SUCCESS;
85 err = meshx_create_plat_composition(&p_dev->composition);
86 if (err)
87 {
88 MESHX_LOGE(MODULE_ID_COMMON, "Failed to create platform composition: (%d)", err);
89 return err;
90 }
91
92 /* Initialize root model */
93 err = meshx_plat_add_element_to_composition(
94 0,
95 p_dev->elements,
96 get_root_sig_models(),
97 NULL,
98 (uint8_t) get_root_sig_models_count(),
99 (uint8_t) ROOT_MODEL_VEN_CNT
100 );
101 if(err)
102 {
103 MESHX_LOGE(MODULE_ID_COMMON, "Failed to add element to composition: (%d)", err);
104 return err;
105 }
106 p_dev->element_idx++;
107 /* Root to be used with fixed format */
108
109 err = meshx_create_element_composition(p_dev, config);
110 if(err)
111 {
112 MESHX_LOGE(MODULE_ID_COMMON, "Failed to create BLE Mesh Element Composition: (%d)", err);
113 return err;
114 }
115 err = meshx_plat_composition_init(
116 p_dev->composition,
117 p_dev->elements,
118 config->cid,
119 config->pid,
120 (uint16_t)p_dev->element_idx
121 );
122 if(err)
123 {
124 MESHX_LOGE(MODULE_ID_COMMON, "Failed to initialise MeshX Composition: (%d)", err);
125 return err;
126 }
127
128 return MESHX_SUCCESS;
129}

177static meshx_err_t meshx_ble_mesh_init(meshx_config_t *config)
178{
179 if(config == NULL || config->product_name == NULL)
180 return MESHX_INVALID_ARG;
181
182 meshx_err_t err;
183 g_prov_cfg.node_name = (uint8_t *)config->product_name;
184
185 err = meshx_platform_bt_init(config->meshx_uuid_addr);
186 MESHX_ERR_PRINT_RET("Platform BT init failed", err);
187
188 g_prov_cfg.uuid = config->meshx_uuid_addr;
189 meshx_dev_restore(&g_dev, config);
190
191 err = meshx_element_init(&g_dev, config);
192 MESHX_ERR_PRINT_RET("Failed to initialize BLE Elements", err);
193
194 err = meshx_plat_ble_mesh_init(&g_prov_cfg, g_dev.composition);
195 MESHX_ERR_PRINT_RET("Failed to initialize BLE Mesh stack", err);
196
197 return MESHX_SUCCESS;
198}

228static meshx_err_t meshx_init_boot_timer(void)
229{
230 if(g_dev.meshx_store.node_addr == MESHX_ADDR_UNASSIGNED)
231 {
232 MESHX_LOGI(MODULE_ID_COMMON, "Device not provisioned, not starting boot timer");
233 return MESHX_SUCCESS;
234 }
235 meshx_err_t err = meshx_os_timer_create("boot_timer",
236 FRESHBOOT_TIMEOUT_MS,
237 false,
238 meshx_init_boot_timer_trigger_cb,
239 &g_boot_timer
240 );
241 MESHX_ERR_PRINT_RET("Failed to create boot timer", err);
242
243 err = meshx_os_timer_start(g_boot_timer);
244 MESHX_ERR_PRINT_RET("Failed to start boot timer", err);
245
246 return err;
247}

258meshx_err_t meshx_init(meshx_config_t const *config)
259{
260 /* Check if the configuration is valid */
261 if(!config)
262 return MESHX_INVALID_ARG;
263
264 meshx_err_t err = MESHX_SUCCESS;
265
266 /* Copy the configuration to the global config structure */
267 memcpy(&g_config, config, sizeof(meshx_config_t));
268
269 meshx_logging_t logging_cfg;
270
271 logging_cfg.def_log_level = config->meshx_log_level == MESHX_LOG_VERBOSE ?
272 CONFIG_MESHX_DEFAULT_LOG_LEVEL : config->meshx_log_level;
273
274 err = meshx_logging_init(&logging_cfg);
275 MESHX_ERR_PRINT_RET("Logging init failed", err);
276
277 /* Initialise Platform deps */
278 err = meshx_platform_init();
279 MESHX_ERR_PRINT_RET("Platform init failed", err);
280
281 /* Initialize OS timer */
282 err = meshx_os_timer_init();
283 MESHX_ERR_PRINT_RET("OS Timer Init failed", err);
284
285 /* Initialize MeshX NVS */
286 err = meshx_nvs_init();
287 MESHX_ERR_PRINT_RET("MeshX NVS Init failed", err);
288
289 /* Initialize application tasks */
290 err = meshx_tasks_init(&g_dev);
291 MESHX_ERR_PRINT_RET("Tasks initialization failed", err);
292
293 /* Register application element callback */
294 err = meshx_app_reg_element_callback(g_config.app_element_cb);
295 MESHX_ERR_PRINT_RET("Failed to register app element callback", err);
296
297 /* Register application control callback */
298 err = meshx_app_reg_system_events_callback(g_config.app_ctrl_cb);
299 MESHX_ERR_PRINT_RET("Failed to register app control callback", err);
300
301 /* Initialize the Bluetooth Mesh Subsystem */
302 err = meshx_ble_mesh_init(&g_config);
303 MESHX_ERR_PRINT_RET("Bluetooth mesh init failed", err);
304
305 /* Initialize boot timer */
306 err = meshx_init_boot_timer();
307 MESHX_ERR_PRINT_RET("Boot Timer Init failed", err);
308 /* Print the MeshX banner */
309 CONFIG_MESHX_LOG_PRINTF(LOG_ANSI_COLOR_REGULAR(LOG_ANSI_COLOR_CYAN) "%s" LOG_ANSI_COLOR_RESET, meshX_banner);
310
311#if CONFIG_ENABLE_UNIT_TEST
312 /* Register unit test command */
313 err = register_ut_command();
314 MESHX_ERR_PRINT_RET("Failed to register unit test command", err);
315
316 /* Initialize unit test console */
317 err = init_unit_test_console();
318 MESHX_ERR_PRINT_RET("Failed to initialize production console", err);
319#endif /* CONFIG_ENABLE_UNIT_TEST */
320
321 return err;
322}
