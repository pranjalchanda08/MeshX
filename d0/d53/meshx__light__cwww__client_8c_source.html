356 .app_idx = el_ctx->app_id,
357 .state = el_ctx->state.on_off,

370 {
371 el_ctx->prev_state.on_off = el_ctx->state.on_off;
372 el_ctx->state.on_off = !el_ctx->state.on_off;
373 }
