65static meshx_err_t meshx_state_change_notify(meshx_lighting_server_cb_param_t *param)
66{
67 meshx_light_ctl_srv_t ctl_srv_params = {
68 .model = param->model};
69
70 if (MESHX_ADDR_IS_UNICAST(param->ctx.dst_addr) || (MESHX_ADDR_BROADCAST(param->ctx.dst_addr)) || (MESHX_ADDR_IS_GROUP(param->ctx.dst_addr) && (MESHX_SUCCESS == meshx_is_group_subscribed(param->model.p_model, param->ctx.dst_addr))))
71 {
72 switch (param->ctx.opcode)
73 {
74 case MESHX_MODEL_OP_LIGHT_CTL_SET:
75 case MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK:
76 ctl_srv_params.state.delta_uv = param->state_change.ctl_set.delta_uv;
77 ctl_srv_params.state.lightness = param->state_change.ctl_set.lightness;
78 ctl_srv_params.state.temperature = param->state_change.ctl_set.temperature;
79 break;
80 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET:
81 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET:
82 ctl_srv_params.state.delta_uv = param->state_change.ctl_temp_set.delta_uv;
83 ctl_srv_params.state.temperature = param->state_change.ctl_temp_set.temperature;
84 break;
85 default:
86 return MESHX_NOT_SUPPORTED;
87 }
88 /* Send msg for hw manipulation */
89 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "HW change requested, Element_id: 0x%x",
90 param->model.el_id);
91
92 meshx_err_t err = control_task_msg_publish(
93 CONTROL_TASK_MSG_CODE_EL_STATE_CH,
94 CONTROL_TASK_MSG_EVT_EL_STATE_CH_SET_CTL,
95 &ctl_srv_params,
96 sizeof(meshx_light_ctl_srv_t));
97 return err;
98 }
99 return MESHX_NOT_SUPPORTED;
100}

117static meshx_err_t meshx_handle_light_ctl_msg(const dev_struct_t *pdev,
118 const control_task_msg_evt_t evt,
119 meshx_lighting_server_cb_param_t *param)
120{
121 if (!pdev || (evt != MESHX_MODEL_ID_LIGHT_CTL_SRV && evt != MESHX_MODEL_ID_LIGHT_CTL_SETUP_SRV))
122 return MESHX_INVALID_ARG;
123 meshx_err_t err = MESHX_SUCCESS;
124
125 bool send_reply_to_src = false;
126 bool state_change_notify = false;
127 uint32_t op_code = param->ctx.opcode;
128 uint32_t status_op = 0;
129 switch (op_code)
130 {
131 case MESHX_MODEL_OP_LIGHT_CTL_STATUS:
132 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS:
133 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS:
134 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS:
135 /* Ignore, as these are status messages */
136 break;
138 case MESHX_MODEL_OP_LIGHT_CTL_GET:
139 case MESHX_MODEL_OP_LIGHT_CTL_SET:
140 case MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK:
141 status_op = MESHX_MODEL_OP_LIGHT_CTL_STATUS;
142 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_GET)
143 state_change_notify = true;
144
145 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_SET_UNACK)
146 send_reply_to_src = true;
147
148 break;
149 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET:
150 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET:
151 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET_UNACK:
152 status_op = MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_STATUS;
153 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_GET)
154 state_change_notify = true;
155
156 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_SET_UNACK)
157 send_reply_to_src = true;
158 break;
160 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET:
161 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_GET:
162 case MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET_UNACK:
163 status_op = MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_STATUS;
164 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_GET)
165 MESHX_DO_NOTHING;
166
167 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_DEFAULT_SET_UNACK)
168 send_reply_to_src = true;
169
170 break;
171 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET:
172 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET:
173 case MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK:
174 status_op = MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_STATUS;
175 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_GET)
176 MESHX_DO_NOTHING;
177
178 if (op_code != MESHX_MODEL_OP_LIGHT_CTL_TEMPERATURE_RANGE_SET_UNACK)
179 send_reply_to_src = true;
180
181 break;
182 default:
183 MESHX_LOGW(MODULE_ID_MODEL_SERVER, "CTL Unhandled Event %p", (void *)param->ctx.opcode);
184 break;
185 }
186 if (state_change_notify)
187 {
188 err = meshx_state_change_notify(param);
189 if (err)
190 return err;
191 }
192 if (send_reply_to_src
193 /* This is meant to notify the respective publish client */
194 || param->ctx.src_addr != param->model.pub_addr)
195 {
196 /* Here the message was received from unregistered source and mention the state to the respective client */
197 MESHX_LOGD(MODULE_ID_MODEL_SERVER, "PUB: src|pub %x|%x", param->ctx.src_addr, param->model.pub_addr);
198 param->ctx.opcode = (uint16_t)status_op;
199 param->ctx.dst_addr = param->model.pub_addr;
200
201 err = meshx_gen_light_srv_send_msg_to_ble(
202 CONTROL_TASK_MSG_EVT_TO_BLE_SET_CTL_SRV,
203 param
204 );
205 }
206
207 return err;
208}

391 meshx_light_ctl_setup_server_delete();

393 meshx_light_ctl_server_delete(p_model);
394 return err;
395 }
